import { type Params, generatePath, matchRoutes, useMatches } from 'react-router-dom'
import Box from '@mui/material/Box'

import { DocumentationSupportLinks } from '@cloudbees/shared-components/DocumentationSupportLinks'
import { ProductEmblem } from '@cloudbees/shared-components/ProductEmblem'
import { type ISidebarItem, GlobalSidebar } from '@cloudbees/shared-patterns/GlobalSidebar'
import { PrimarySkeleton } from '@cloudbees/shared-patterns/PrimarySkeleton'
import { SidebarLayout } from '@cloudbees/shared-patterns/SidebarLayout'
import { type IMatch, type SimpleRouteInjection } from '@cloudbees/shared-types'
import { useParams } from '@cloudbees/shared-utils/navigation'

import { UserBar } from '../components'

import { useRoutesContext } from './RouteProvider'

const getSidebarItems = (
  matches: IMatch[],
  availableRoutes: SimpleRouteInjection[],
  params: Params,
  sidebarType: 'primarySidebarItems' | 'secondarySidebarItems',
): ISidebarItem[] => {
  const sidebarItems: ISidebarItem[] = []

  // 1. For each route, gather the sidebar items defined in the handle
  matches.forEach(match => {
    const sidebarHandleItems = match.handle?.[sidebarType] || []

    const subSidebarItems: ISidebarItem[] = []

    sidebarHandleItems.forEach(sidebar => {
      const normalizedPath = match.pathname === '/' ? '' : match.pathname
      const generatedPath = `${generatePath(sidebar.path, params)}`

      const matchedRoute = matchRoutes(availableRoutes, generatedPath)

      // If the route is not defined, don't add it to the sidebar
      // If a route matches a wildcard, don't add it to the sidebar
      if (!matchedRoute || matchedRoute.length === 0 || !!matchedRoute[0].params['*']) {
        return
      }

      const component = (
        <SidebarLayout.SidebarItem
          href={generatedPath}
          key={generatedPath}
          textContent={sidebar.label}
        >
          {normalizedPath === '' ? sidebar.label : <Box sx={{ pl: 7 }}>{sidebar.label}</Box>}
        </SidebarLayout.SidebarItem>
      )

      subSidebarItems.push({
        generatedPath,
        component,
      })
    })

    // 2. Find the index of the parent sidebar.  A parent sidebar has a generatedPath that
    // is a substring of the generatedPath of one of the subSidebarItems.
    const foundIndex = sidebarItems.findIndex(element => {
      return subSidebarItems[0]?.generatedPath.includes(element.generatedPath)
    })

    // 3. Insert the subSidebarItems into the sidebarItems array after the parent route
    sidebarItems.splice(
      foundIndex > -1 ? foundIndex + 1 : sidebarItems.length - 1,
      0,
      ...subSidebarItems,
    )
  })

  return sidebarItems
}

/**
 *
 * "MainLayout" component uses extension points to discover available
 * routes within the app, and then utilizes the route matching
 * functionality from react-router-v6 in order to generate
 * relevant and contextual sidebar links.
 *
 * To learn how to structure your routes in a manner that
 * can be utilized by this sidebar, refer to the "appRoutes" object in the
 * "RouteProvider.tsx" file and SimpleRouteInjection documentation.
 *
 * Only defined routes show up in sidebar.
 */
export const MainLayout = () => {
  const matches = useMatches() as IMatch[]
  const allAvailableRoutes = useRoutesContext()
  const params = useParams()

  const topSidebarItems = getSidebarItems(
    matches,
    allAvailableRoutes,
    params,
    'primarySidebarItems',
  )

  return (
    <PrimarySkeleton
      globalSidebar={
        <GlobalSidebar
          header={<ProductEmblem title="CloudBees" subtitle="Platform Admin Tools" link="/" />}
          topSidebarItems={topSidebarItems}
        />
      }
      appBarItems={[
        // in admin-tools, we should always go to the CB documentation
        <DocumentationSupportLinks key="doc-links" orgIsPaid />,
        <UserBar key="user-bar" />,
      ]}
    />
  )
}

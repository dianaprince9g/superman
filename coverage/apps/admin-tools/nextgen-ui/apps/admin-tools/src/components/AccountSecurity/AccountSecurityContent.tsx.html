import { useTranslation } from 'react-i18next'
import Card from '@mui/material/Card'
import CardActions from '@mui/material/CardActions'
import CardContent from '@mui/material/CardContent'
import Grid from '@mui/material/Grid'
import Typography from '@mui/material/Typography'
 
import { EMPTY_RECORD } from '@cloudbees/api-platform/queries/constants'
import { useKeycloakEnabled } from '@cloudbees/shared-auth/AuthProvider/useKeycloakEnabled'
import { useGetUserData } from '@cloudbees/shared-network/useGetUserData'
import { useAppContext } from '@cloudbees/shared-providers/components/AppContextProvider'
import { useParams } from '@cloudbees/shared-utils/navigation'
import { ResetPasswordSection } from '@cloudbees/user-management/AccountSecurity'
import { TwoFactorAuthenticationTab } from '@cloudbees/user-management/multiFactorAuth/TwoFactorAuthenticationTab'
import { useMfaQuery } from '@cloudbees/user-management/ProfilePage/useMfaQuery'
 
export const RESET_PASSWORD_SUCCESS = `The email with the reset password link was sent to`
export const RESET_PASSWORD_ERROR =
  'An error occurred when sending the reset password link. Try again later.'
 
export const AccountSecurityContent = () => {
  const { t } = useTranslation('user-management')
 
  const user = useGetUserData()
  const appContext = useAppContext()
  const keycloakEnabled = useKeycloakEnabled()
 
  const { userId } = useParams()
 
  const { data: keycloakMfaData, isLoading } = useMfaQuery({
    userId: userId ?? null,
    enabled: keycloakEnabled,
  })
 
  if (!user || isLoading) {
    return <div>Loading...</div>
  }
 
  let getMfaHelpMessage
  if (appContext === 'admin-tools') {
    let mfaEnabled
    if (keycloakEnabled) {
      mfaEnabled = keycloakMfaData && keycloakMfaData?.id && keycloakMfaData?.id !== EMPTY_RECORD
    } else {
      mfaEnabled = !!user?.metadata?.use_mfa
    }
    if (mfaEnabled) {
      getMfaHelpMessage = t('user-security.admin_tools_mfa_help_message_for_when_mfa_is_enabled')
    } else {
      getMfaHelpMessage = t(
        'user-security.admin_tools_mfa_help_message_for_when_mfa_is_already_disabled',
      )
    }
  }
 
  return (
    <Grid container rowSpacing={1} columnSpacing={{ xs: 1, sm: 2, md: 3 }} sx={{ mt: 2 }}>
      {appContext === 'admin-tools' && (
        <Grid item xs={12} sx={{ mb: 2 }}>
          <Typography variant="h4">{t('user-security.security-section')}</Typography>
        </Grid>
      )}
      <ResetPasswordSection email={user.email} loginProvider={user.logins[0].provider} />
      <Grid item xs={12}>
        <Card variant="outlined" style={{ height: '100%' }}>
          <CardContent>
            <Typography variant="h5" component="div" data-testid="mfa-card-title">
              {t('user-security.mfa-title')}
            </Typography>
            <br />
            <Typography variant="body1" data-testid="mfa-card-description">
              {t('user-security.mfa-description')}
            </Typography>
            <br />
            {getMfaHelpMessage && (
              <Typography variant="body1" data-testid="mfa-help-description">
                {getMfaHelpMessage}
              </Typography>
            )}
          </CardContent>
          <CardActions>
            <TwoFactorAuthenticationTab />
          </CardActions>
        </Card>
      </Grid>
    </Grid>
  )
}

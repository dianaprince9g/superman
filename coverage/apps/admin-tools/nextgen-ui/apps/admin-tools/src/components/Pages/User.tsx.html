import { useTranslation } from 'react-i18next'
import Box from '@mui/material/Box'
import Divider from '@mui/material/Divider'
import Stack from '@mui/material/Stack'
import TextField from '@mui/material/TextField'
import Typography from '@mui/material/Typography'
import { createColumnHelper } from '@tanstack/react-table'

import { useApiMutation } from '@cloudbees/api-platform/queries/guiderails'
import { WriteActionModal } from '@cloudbees/organization-management/WriteAction/WriteActionModal'
import {
  type IWriteActionModalState,
  WriteActionModalAPI,
} from '@cloudbees/organization-management/WriteAction/WriteActionModalAPI'
import { JsonViewer } from '@cloudbees/shared-components/JsonViewer'
import { useGetUserData } from '@cloudbees/shared-network/useGetUserData'
import { Filter, useFilterContext } from '@cloudbees/shared-patterns/Filter'
import { PageHeader } from '@cloudbees/shared-patterns/PageHeader'
import { TableNew } from '@cloudbees/shared-patterns/TableNew'
import { useFeatureFlagEnabled } from '@cloudbees/shared-rox/useFeatureFlagEnabled'

import { AccountSecurityContent } from '../AccountSecurity/AccountSecurityContent'

import { Organizations } from './UsersPage/Organizations'

const makeHeader = (title: string) => () => <span>{title}</span>

const UserTable = () => {
  const { t } = useTranslation('user-profile')
  const { data } = useFilterContext()
  const columnHelper = createColumnHelper<{ provider: string; id: string }>()
  const columns = [
    columnHelper.accessor('provider', {
      header: makeHeader(t('login_provider')),
      enableSorting: false,
    }),
    columnHelper.accessor('id', {
      header: makeHeader(t('id')),
      enableSorting: false,
    }),
  ]
  return <TableNew data={data} columns={columns} />
}

export const UserPage = () => {
  const { t } = useTranslation('user-profile')
  const user = useGetUserData()

  const { POST } = useApiMutation()
  const modalState: IWriteActionModalState = WriteActionModalAPI.use()

  const showPublishButtons = useFeatureFlagEnabled('platform', 'showPublishButtons')

  if (!user) {
    return <div>Loading...</div>
  }

  const email = user.email || user.logins[0]?.email

  return (
    <>
      {modalState.isOpen && <WriteActionModal />}
      <PageHeader.Wrapper>
        <PageHeader.Breadcrumbs>
          <PageHeader.BreadcrumbItem>Overview</PageHeader.BreadcrumbItem>
          <PageHeader.BreadcrumbItem href="/users">Users</PageHeader.BreadcrumbItem>
          <PageHeader.BreadcrumbItem>
            {`${user.firstName} ${user.lastName}`}
          </PageHeader.BreadcrumbItem>
          <PageHeader.BreadcrumbItem>{t('breadcrumbs.user_profile')}</PageHeader.BreadcrumbItem>
        </PageHeader.Breadcrumbs>
        <PageHeader.Title>{t('title')}</PageHeader.Title>
        <PageHeader.Description>{`${user.firstName} ${user.lastName}`}</PageHeader.Description>
        <PageHeader.Actions>
          {showPublishButtons && (
            <PageHeader.ActionItem
              variant="contained"
              onClick={() => {
                WriteActionModalAPI.setupModal({
                  writeAction: async () => {
                    return await POST({
                      path: `v1/users/${user.id}`,
                      body: {
                        givenname: user.firstName,
                        familyname: user.lastName,
                      },
                    })
                  },
                  onClose: () => WriteActionModalAPI.closeModal(),
                  title: t('publish_user_modal_title'),
                  successMessage: t('publish_user_success'),
                  failureMessage: t('publish_user_failure'),
                })
                WriteActionModalAPI.openModal()
              }}
            >
              {t('publish_user')}
            </PageHeader.ActionItem>
          )}
        </PageHeader.Actions>
      </PageHeader.Wrapper>
      <Stack gap={3} sx={{ mt: 3, mx: 2 }}>
        <TextField fullWidth label={t('id')} value={user.id} disabled />
        <TextField fullWidth label={t('first_name')} value={user.firstName} disabled />
        <TextField fullWidth label={t('last_name')} value={user.lastName} disabled />
        <TextField
          fullWidth
          value={email}
          disabled
          label={
            <label>
              {t('email')}
              <Typography component="span" color={user.isVerified ? 'success.main' : 'error.main'}>
                {user.isVerified ? t('email_verified') : t('email_not_verified')}
              </Typography>
            </label>
          }
        />
        <Typography variant="h5" sx={{ mt: 2, mx: 2 }}>
          {t('logins')}
        </Typography>
      </Stack>
      <Filter.Container data={user.logins || []} density={'compact'}>
        <UserTable />
      </Filter.Container>
      <Box sx={{ mt: 4, mx: 2 }}>
        <JsonViewer
          src={user}
          id="user-json"
          collapsedText={t('show_user_json')}
          expandedText={t('hide_user_json')}
        />
      </Box>
      <Box sx={{ mx: 2 }}>
        <Divider sx={{ mt: 4 }} />
        <AccountSecurityContent />
      </Box>
      <Organizations user={user} />
    </>
  )
}

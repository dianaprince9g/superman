import { type MouseEvent, useState } from 'react'
import { useTranslation } from 'react-i18next'
import Check from '@mui/icons-material/Check'
import CorporateFareIcon from '@mui/icons-material/CorporateFare'
import Box from '@mui/material/Box'
import Button from '@mui/material/Button'
import Divider from '@mui/material/Divider'
import ListItemIcon from '@mui/material/ListItemIcon'
import ListItemText from '@mui/material/ListItemText'
import ListSubheader from '@mui/material/ListSubheader'
import Menu from '@mui/material/Menu'
import MenuItem from '@mui/material/MenuItem'
import Typography from '@mui/material/Typography'
 
import { ROUTE_ORG_PREFIX } from '@cloudbees/route-config-nextgen'
import { type IOrganization } from '@cloudbees/shared-auth/types'
import { Link, useNavigate } from '@cloudbees/shared-utils/navigation'
import { useSwitchOrg } from '@cloudbees/user-management/hooks/useSwitchOrg'
 
export const UserOrgsBar = () => {
  const { t } = useTranslation('nextgen-ui', {
    keyPrefix: 'user-bar',
  })
 
  const navigate = useNavigate()
  const [anchorEl, setAnchorEl] = useState<null | HTMLElement>(null)
  const open = Boolean(anchorEl)
  const { currentOrg, currentOrgName, switchOrg, userOrgs } = useSwitchOrg()
 
  const handleClick = (event: MouseEvent<HTMLElement>) => {
    setAnchorEl(event.currentTarget)
  }
 
  const handleClose = (value: MouseEvent<HTMLElement>) => {
    setAnchorEl(null)
    const href = value.currentTarget.getAttribute('href')
    if (href) {
      navigate(href)
    }
  }
 
  const createOrgUrl = (orgDomainName: string): string => {
    const encodedOrgDomain = encodeURIComponent(orgDomainName)
    return `${ROUTE_ORG_PREFIX}/${encodedOrgDomain}`
  }
 
  const handleMenuItemClick = (orgDomainName: string) => {
    switchOrg(orgDomainName)
  }
 
  return (
    <Box>
      <Button
        onClick={handleClick}
        data-testid="orgs-menu-cta"
        sx={{
          display: 'flex',
          alignItems: 'center',
          gap: 2,
          paddingLeft: 1,
          paddingRight: 1,
          color: 'inherit',
        }}
      >
        <Typography variant="body2" noWrap>
          {currentOrgName}
        </Typography>
        <CorporateFareIcon />
      </Button>
      <Menu anchorEl={anchorEl} onClose={handleClose} open={open} data-testid="orgs-menu">
        <ListSubheader>{t('organization')}</ListSubheader>
        <MenuItem data-testid="current-org-name">{currentOrgName}</MenuItem>
        <Divider />
        <ListSubheader>{t('recent-organizations')}</ListSubheader>
        {userOrgs?.userOrgsList.slice(0, 5).map((org: IOrganization) => {
          return (
            <MenuItem
              key={org.id}
              component={Link}
              to={createOrgUrl(org.domain)}
              onClick={() => handleMenuItemClick(org.domain)}
              data-testid="org-item"
            >
              <ListItemText>{org.name}</ListItemText>
              {org.id === currentOrg.id && (
                <ListItemIcon>
                  <Check />
                </ListItemIcon>
              )}
            </MenuItem>
          )
        })}
        <Divider />
        <MenuItem onClick={handleClose} component={Link} to="/my-account">
          {t('show-all-organizations')}
        </MenuItem>
      </Menu>
    </Box>
  )
}

import { Navigate, Outlet, useMatches } from 'react-router-dom'
 
import { useGetAccounts } from '@cloudbees/api-platform/queries/enforcement/useGetAccounts'
import { useEntitlementContext } from '@cloudbees/shared-network/EntitlementsProvider'
import { useFeatureFlagContext } from '@cloudbees/shared-rox/FeatureFlagProvider'
import { type IMatch } from '@cloudbees/shared-types'
import { useParams } from '@cloudbees/shared-utils/navigation/hooks'
 
// Redirect to home if route is missing any required feature flags, entitlements, params or multi-tenant only
export const ValidRouteOutlet = () => {
  const { data } = useGetAccounts()
  const matches = useMatches() as IMatch[]
  const params = useParams()
  const featureFlags = useFeatureFlagContext()
  const entitlements = useEntitlementContext()
 
  const requiredFeatureFlags = matches.map(match => match.handle?.featureFlag)
 
  // Navigate routes that are missing the required feature flag
  const missingARequiredFeatureFlag = requiredFeatureFlags.some(featureFlag => {
    if (!featureFlag) return false
 
    const { name, namespace } = featureFlag
    const flag = featureFlags[namespace]?.[name]
 
    return !flag?.isEnabled()
  })
 
  const requiredEntitlements = matches.map(match => match.handle?.entitlement)
 
  // Navigate routes that are missing the required entitlement
  const missingARequiredEntitlement = requiredEntitlements.some(
    entitlementName => entitlementName && !entitlements.includes(entitlementName),
  )
 
  const requiredParams = matches.map(match => match.handle?.requiredFilter)
 
  // Navigate routes that are missing the required params
  const missingARequiredParam = requiredParams.some(
    requiredParam => requiredParam && !params[requiredParam],
  )
 
  const missingMultiTenant = data?.isVpc && matches.some(match => match.handle?.multiTenantOnly)
 
  if (
    missingARequiredFeatureFlag ||
    missingARequiredEntitlement ||
    missingARequiredParam ||
    missingMultiTenant
  ) {
    return <Navigate to={`/orgs/${params.orgId}`} replace />
  }
 
  return <Outlet />
}

import { useCallback, useState } from 'react'
import FilterIcon from '@mui/icons-material/FilterAlt'
import Box from '@mui/material/Box'
import Button from '@mui/material/Button'
import Grid from '@mui/material/Grid'
import InputLabel from '@mui/material/InputLabel'
import Popover from '@mui/material/Popover'
import Stack from '@mui/material/Stack'
import Tooltip from '@mui/material/Tooltip'
import i18next from 'i18next'
 
import { useColorMode } from '@cloudbees/shared-theme/UIProvider'
 
import { DURATION_SELECT, MULTI_SELECT, SELECT } from '../context/constants'
import { useDashboardFilterContext } from '../context/DashboardFilterContextProvider'
import { type IDashboardFilter } from '../DashboardFilterContainer/types'
 
import { DashboardFilter } from './../DashboardFilter'
 
const t = (key: string) => i18next.t(`filters.${key}`, { ns: 'shared-patterns' })
 
const AppliedFilters = () => {
  const { appliedFilters } = useDashboardFilterContext()
  return (
    <Box sx={{ pl: 2 }}>
      <InputLabel>{t('appliedFilters')}</InputLabel>
      {Object.keys(appliedFilters).length > 0 && (
        <Box sx={{ display: 'flex' }}>
          {Object.keys(appliedFilters).map((appliedFilter, index) => {
            const optionText = appliedFilters[appliedFilter]?.optionText?.toLocaleString()
            return (
              <InputLabel key={`appliedFilter-${index}`} sx={{ pr: 2, maxWidth: '500px' }}>
                {`${appliedFilters[appliedFilter].title}: `}
                <Tooltip title={optionText}>
                  <b>{optionText}</b>
                </Tooltip>
              </InputLabel>
            )
          })}
        </Box>
      )}
    </Box>
  )
}
 
const FilterContent = ({ filters }: { filters: IDashboardFilter[] }) => {
  const renderContentFilter = useCallback(
    (filter: IDashboardFilter, index: number) => (
      <Grid item key={`content-${index}`} pb={2} pt={1}>
        {Filters(filter)}
      </Grid>
    ),
    [filters],
  )
  return <>{filters?.length > 0 && filters?.filter(x => !x?.hide).map(renderContentFilter)}</>
}
 
const Filters = (filter: IDashboardFilter) => {
  const { type } = filter
  const filterTypes = {
    [SELECT]: () => DashboardFilter.Select(filter),
    [MULTI_SELECT]: () => DashboardFilter.MultiSelect(filter),
    [DURATION_SELECT]: () => DashboardFilter.DurationSelect(filter),
    default: () => null,
  }
  const renderFilter = filterTypes[type] || filterTypes.default
  return renderFilter()
}
 
export const DashboardFilterTopBar = () => {
  const [anchorEl, setAnchorEl] = useState<null | HTMLElement>(null)
  const { filters, applyFilters, resetFilters, handleFilterClose } = useDashboardFilterContext()
  const { mode } = useColorMode()
 
  const handleClick = (event: React.MouseEvent<HTMLElement>) => {
    setAnchorEl(event.currentTarget)
  }
 
  const handleApplyClick = () => {
    setAnchorEl(null)
    applyFilters()
  }
 
  const handleClose = () => {
    setAnchorEl(null)
    handleFilterClose()
  }
 
  return (
    <Stack direction="row" justifyContent="space-between" alignItems="center" padding={2}>
      <Box sx={{ display: 'flex' }}>
        <Button size="medium" variant="contained" startIcon={<FilterIcon />} onClick={handleClick}>
          {t('title')}
        </Button>
        {AppliedFilters()}
      </Box>
      <Popover
        keepMounted
        open={Boolean(anchorEl)}
        anchorEl={anchorEl}
        onClose={handleClose}
        anchorOrigin={{
          vertical: 50,
          horizontal: 'left',
        }}
      >
        <Box
          width={350}
          padding={3}
          sx={{ backgroundColor: mode === 'dark' ? '#191919' : '#fafafa' }}
        >
          <FilterContent filters={filters} />
          <Box sx={{ display: 'flex', justifyContent: 'space-between' }}>
            <Button variant="text" onClick={resetFilters}>
              {t('resetFilters')}
            </Button>
            <Button variant="contained" onClick={handleApplyClick}>
              {t('apply')}
            </Button>
          </Box>
        </Box>
      </Popover>
    </Stack>
  )
}

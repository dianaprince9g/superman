export type PluginIconProps = {
  name: string
  size?: number
  className?: string
}
 
export const PluginIcon = ({ name, className, size = 18, ...others }: PluginIconProps) => {
  const imageSource = pluginImageMappings[name] ?? DEFAULT_PLUGIN_TOOL
  const isInverted = invertedDarkModeImages.indexOf(name) !== -1
  const theme = useTheme()
 
  return (
    <img
      className={className}
      style={{ filter: isInverted && theme.palette.mode === 'dark' ? 'invert(1)' : '' }}
      alt={name ?? 'Plugin Logo'}
      width={size}
      height={size}
      {...others}
      src={imageSource}
    />
  )
}

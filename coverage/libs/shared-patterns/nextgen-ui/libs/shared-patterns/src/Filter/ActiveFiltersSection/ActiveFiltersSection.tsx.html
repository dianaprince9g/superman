import { useTranslation } from 'react-i18next'
import Box from '@mui/material/Box'
import Button from '@mui/material/Button'
import Chip from '@mui/material/Chip'
import Stack from '@mui/material/Stack'
import Typography from '@mui/material/Typography'
import pluralize from 'pluralize'
 
import { useFilterContext } from '../context'
interface IActiveFiltersSectionProps {
  /**
   * Optional namespace for translation
   * @default ''
   */
  translationsNamespace?: string
  /**
   * Optional key prefix for translation
   * @default ''
   */
  translationsKeyPrefix?: string
}
 
export const ActiveFiltersSection = ({
  translationsNamespace,
  translationsKeyPrefix,
}: IActiveFiltersSectionProps) => {
  const { t: tLocal } = useTranslation('shared-patterns', { keyPrefix: 'filters' })
  const { t } = useTranslation(translationsNamespace, { keyPrefix: translationsKeyPrefix })
  const { activeFilters, clearFilters, updateFilterValue } = useFilterContext()
 
  const emptyActiveFilters = (index: number = 0) => <Box key={index} sx={{ height: '60px' }} />
 
  const renderChip = (label: string, index: number) => {
    const filterOptions = activeFilters[label].value?.split('+') || []
 
    if (!filterOptions.length) return emptyActiveFilters(index)
 
    return (
      <Stack key={`${label}${index}`} direction="row">
        <Chip
          size={'medium'}
          label={
            <Typography variant="body1" sx={{ px: 0.5 }}>
              {`${filterOptions.length} ` + pluralize(t(`${label}`), filterOptions.length)}
            </Typography>
          }
          onDelete={() => updateFilterValue(label, '')}
        ></Chip>
      </Stack>
    )
  }
 
  if (!Object.keys(activeFilters).length) return emptyActiveFilters()
 
  return (
    <Stack
      sx={{ px: 2, pt: 2, pb: 1 }}
      direction="row"
      gap={1}
      justifyContent="start"
      data-testid="active-filters-section"
    >
      <Typography sx={{ pt: 1 }} variant="body1">
        {tLocal('active_filters')}
      </Typography>
 
      <Stack direction="row" gap={1} sx={{ pt: 0.5 }}>
        {Object.keys(activeFilters).map((label, index) => renderChip(label, index))}
      </Stack>
 
      <Button variant="text" onClick={clearFilters} data-testid="clear-all-filters">
        {tLocal('clear_all')}
      </Button>
    </Stack>
  )
}

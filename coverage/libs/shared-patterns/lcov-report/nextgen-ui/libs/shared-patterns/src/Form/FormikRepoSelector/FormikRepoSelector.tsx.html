import { useEffect, useRef, useState } from 'react'
import { useTranslation } from 'react-i18next'
import Box from '@mui/material/Box'
import CircularProgress from '@mui/material/CircularProgress'
import RadioGroup from '@mui/material/RadioGroup'
import Stack from '@mui/material/Stack'
import { useField } from 'formik'
 
import { Filter, useFilterContext } from '../../Filter'
 
import { RepoSelectorItem } from './RepoSelectorItem/RepoSelectorItem'
import { type IConnectedStackProps, type IRepoSelectorProps } from './types'
 
export const FormikRepoSelector = ({
  options,
  height = 400,
  id,
  name,
  onChange,
  isLoading,
}: IRepoSelectorProps) => {
  const { t } = useTranslation('pipeline-management', {
    keyPrefix: 'create-component-drawer.select-repository-step',
  })
  return (
    <Filter.Container data={options} disableQueryString>
      <Stack direction="column" gap={2}>
        <Filter.Search autoFocus attributes={['name', 'org', 'type', 'value']} />
        <Filter.Toggle
          attribute="isSelectable"
          initialValue={true}
          label={t('filter-disabled-repositories') || undefined}
        />
        <ConnectedStack
          height={height}
          id={id}
          name={name}
          onChange={onChange}
          isLoading={isLoading}
        />
      </Stack>
    </Filter.Container>
  )
}
 
const ITEMS_PER_PAGE = 25
 
const ConnectedStack = ({ height, id, name, onChange, isLoading }: IConnectedStackProps) => {
  const { data } = useFilterContext()
  const [field, , helpers] = useField(id)
  const [loadedItems, setLoadedItems] = useState(data.slice(0, ITEMS_PER_PAGE))
  const [pageNum, setPageNum] = useState(1)
  const [lastElement, setLastElement] = useState()
 
  const { t } = useTranslation('pipeline-management', {
    keyPrefix: 'create-component-drawer.select-repository-step',
  })
 
  const observer = useRef(
    new IntersectionObserver(entries => {
      const first = entries[0]
      if (first.isIntersecting) {
        setPageNum((no: number) => no + 1)
      }
    }),
  )
 
  const loadMoreItems = async () => {
    const nextItemToLoadIndex = ITEMS_PER_PAGE * (pageNum - 1)
    const itemsToLoad = data.slice(nextItemToLoadIndex, nextItemToLoadIndex + ITEMS_PER_PAGE)
    // setTimeout to fake backend interaction :>
    setTimeout(() => setLoadedItems([...loadedItems, ...itemsToLoad]), 100)
  }
 
  const TOTAL_PAGES =
    data.length % ITEMS_PER_PAGE > 0
      ? Math.ceil(data.length / ITEMS_PER_PAGE)
      : data.length / ITEMS_PER_PAGE
 
  // reset loaded items in case data changes (reftech or filtering)
  useEffect(() => {
    const itemsToLoad = isLoading || !data.length ? [] : data.slice(0, ITEMS_PER_PAGE)
    setLoadedItems(itemsToLoad)
    setPageNum(1)
  }, [data])
 
  useEffect(() => {
    if (pageNum > 1 && pageNum <= TOTAL_PAGES) {
      loadMoreItems()
    }
  }, [pageNum])
 
  useEffect(() => {
    const currentElement = lastElement
    const currentObserver = observer.current
 
    if (currentElement) {
      currentObserver.observe(currentElement)
    }
 
    return () => {
      if (currentElement) {
        currentObserver.unobserve(currentElement)
      }
    }
  }, [lastElement])
 
  const handleChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    helpers.setValue(event.target.value)
    onChange?.(event)
  }
 
  const selectorSizeRef = useRef<number>(0)
 
  useEffect(() => {
    const radioGroupHeight = document.querySelector('#radio-group-repo-selector')?.clientHeight || 0
    const currentHeight =
      !!radioGroupHeight && height > radioGroupHeight ? radioGroupHeight : height
 
    selectorSizeRef.current = currentHeight
  }, [height])
 
  return (
    <Box sx={{ height: selectorSizeRef.current, overflowY: 'auto', overflowX: 'hidden' }}>
      {isLoading ? (
        <Box
          sx={{
            display: 'flex',
            alignItems: 'center',
            justifyContent: 'center',
            height: '100%',
            minHeight: '100%',
          }}
        >
          <CircularProgress color="primary" />
        </Box>
      ) : (
        <RadioGroup
          name={name || 'repository-selector'}
          onChange={handleChange}
          value={field.value || ''}
          defaultValue={''}
          id={'radio-group-repo-selector'}
        >
          <Stack direction="column" gap={1}>
            {loadedItems.length > 0 &&
              loadedItems.map((item, i) => {
                return i === loadedItems.length - 1 && pageNum <= TOTAL_PAGES ? (
                  <div key={item.value} ref={setLastElement as any}>
                    <RepoSelectorItem {...item} />
                  </div>
                ) : (
                  <RepoSelectorItem key={item.value} {...item} />
                )
              })}
            {loadedItems.length === 0 && <Box>{t('filter-no-results')}</Box>}
            {/* Always display loader unless we are on the last page (it will be hidden if not scroll down completely anyway) */}
            {pageNum < TOTAL_PAGES && (
              <Box
                sx={{
                  display: 'flex',
                  alignItems: 'center',
                  justifyContent: 'center',
                  height: '100%',
                  minHeight: '100%',
                }}
              >
                <CircularProgress color="primary" />
              </Box>
            )}
          </Stack>
        </RadioGroup>
      )}
    </Box>
  )
}

import {
  type ReactPortal,
  createContext,
  useCallback,
  useContext,
  useEffect,
  useMemo,
  useState,
} from 'react'
import { createPortal } from 'react-dom'
import Alert from '@mui/material/Alert'
import Box from '@mui/material/Box'
import Button from '@mui/material/Button'
import LinearProgress from '@mui/material/LinearProgress'
import Typography from '@mui/material/Typography'
 
import { Drawer } from '../Drawer/Drawer'
 
export interface IDrawerStep {
  element: JSX.Element
  title?: string | ((data: Record<string, string>) => string)
  /**
   * (Optional) Defines a secondary drawer for the step that will be shown at the right side of the main drawer
   */
  secondaryDrawer?: React.ReactNode
}
 
export interface IDrawerStepperProps {
  /**
   * The steps to render in the stepper
   * When title is a function it can dynamically change based on the data
   */
  steps: IDrawerStep[]
 
  /**
   * (Optional) Whether to hide the progress bar
   * @default false
   */
  hideProgress?: boolean
 
  /**
   * Whether to show the drawer
   */
  showDrawer: boolean
 
  /**
   * Callback to close the drawer
   */
  onCloseDrawer: (step?: number) => void
 
  /**
   * The title of the drawer
   */
  title: string
 
  /**
   * The drawer size.
   * @default 's'
   */
  size?: 's' | 'm' | 'l'
 
  /**
   * Callback function that gets triggered after the last step is completed
   * @param data The data collected from the steps
   */
  onFinished?: (data: Record<string, string>) => void
 
  /**
   * (Optional) Whether to show the secondary drawer if the current step has one
   */
  secondaryDrawerOpen?: boolean
 
  /**
   * (Optional) Whether to show an explicit close button (along with the back button)
   * on the last step. This is useful when the last step only displays information.
   */
  showCloseButtonOnLastStep?: boolean
}
 
interface IStepperContext {
  nextStep: (skip?: number) => void
  nextButton: (component: React.ReactNode) => ReactPortal | null
  data: Record<string, string>
  setData: (data: Record<string, string>) => void
  setError: (errorMessage: React.ReactNode) => void
}
 
export const DrawerStepperContext = createContext<IStepperContext>({
  nextStep: () => undefined,
  nextButton: () => null,
  data: {},
  setData: () => undefined,
  setError: () => undefined,
})
 
export const useDrawerStepperContext = () => {
  return useContext(DrawerStepperContext)
}
 
/**
 * A linear progress stepper pattern in a Drawer
 */
export const DrawerStepper = (props: IDrawerStepperProps) => {
  const [currentStep, setCurrentStep] = useState(0)
  // Keeps track of the number of steps to skip
  const [stepStack, setStepStack] = useState<number[]>([])
 
  // contains arbitrary data that can be passed and set between steps
  const [data, setData] = useState<Record<string, string>>({})
  const [error, setError] = useState<React.ReactNode | undefined>()
  const [footerRef, setFooterRef] = useState<HTMLDivElement | null>(null)
 
  const closeDrawer = useCallback(() => {
    // Wait for the drawer to close before going back to the first step
    setTimeout(() => {
      setCurrentStep(0)
      setStepStack([])
      setData({})
      setError(undefined)
    }, 200)
    props.onCloseDrawer(currentStep)
  }, [currentStep, props])
 
  useEffect(() => {
    if (currentStep === props.steps.length) {
      closeDrawer()
      props.onFinished && props.onFinished(data)
    }
  }, [currentStep, data])
 
  /**
   * Go to the next step.  If skip is provided, skip that many steps
   */
  const nextStep = useCallback(
    (skip = 0) => {
      setError(undefined)
      setStepStack(prevStack => [...prevStack, skip + 1])
      setCurrentStep(prevStep => prevStep + skip + 1)
    },
    [currentStep, props.steps.length],
  )
 
  const prevStep = useCallback(() => {
    setError(undefined)
 
    const lastStepSkip = stepStack.at(-1)
 
    if (!lastStepSkip) {
      closeDrawer()
    } else {
      setCurrentStep(prevStep => prevStep - lastStepSkip)
      setStepStack(prevStack => prevStack.slice(0, prevStack.length - 1))
    }
  }, [closeDrawer, stepStack])
 
  const nextButton = useCallback(
    (component: React.ReactNode) => {
      if (footerRef) {
        return createPortal(component, footerRef)
      }

      return null
    },
    [footerRef],
  )
 
  const setDataMemo = useCallback((data: Record<string, string>) => {
    setData(prevData => ({ ...prevData, ...data }))
  }, [])
 
  const contextValue = useMemo(
    () => ({
      nextStep,
      nextButton,
      setData: setDataMemo,
      data,
      setError,
    }),
    [nextStep, nextButton, setDataMemo, data],
  )
 
  // Watch the footer, when it is rendered we can render the step component
  // This ensures the nextButton is always rendered in the footer
  const refCallback = useCallback((node: HTMLDivElement) => {
    if (node !== null) {
      setFooterRef(node)
    }
  }, [])
 
  if (!props.steps[currentStep]) {
    return null
  }
 
  const progressPercent = ((currentStep + 1) / props.steps.length) * 100
  const { element, title, secondaryDrawer } = props.steps[currentStep]
  const stepTitle = typeof title === 'function' ? title(data) : title
 
  return (
    <Drawer.Wrapper
      open={props.showDrawer}
      onClose={closeDrawer}
      data-testid="multi-step-drawer"
      key={props.title}
      size={props.size ?? 's'}
      secondaryDrawer={secondaryDrawer}
      secondaryDrawerOpen={!!secondaryDrawer && props.secondaryDrawerOpen}
    >
      <Drawer.Header title={props.title} onClose={closeDrawer}>
        {stepTitle && <Typography>{stepTitle}</Typography>}
        {!props.hideProgress && (
          <LinearProgress variant="determinate" value={Math.min(progressPercent, 95)} />
        )}
        {error && <Alert severity="error">{error}</Alert>}
      </Drawer.Header>
      <Drawer.Content>
        <Box sx={{ mt: 3 }}>
          {footerRef && (
            <DrawerStepperContext.Provider value={contextValue}>
              {element}
            </DrawerStepperContext.Provider>
          )}
        </Box>
      </Drawer.Content>
      <Drawer.Footer ref={refCallback}>
        <Button variant="text" onClick={prevStep}>
          {currentStep === 0 ? 'Cancel' : 'Back'}
        </Button>
        {props.showCloseButtonOnLastStep && props.steps.length - 1 === currentStep ? (
          <Button variant="text" onClick={closeDrawer} data-testid="close-button-last-step">
            Close
          </Button>
        ) : null}
      </Drawer.Footer>
    </Drawer.Wrapper>
  )
}

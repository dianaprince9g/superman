import { useAuthorizationsQuery } from '@cloudbees/api-platform/queries/useAuthorizationsQuery'
import { useComponentsQuery } from '@cloudbees/api-platform/queries/useComponentsQuery'
import { useGetOrganization } from '@cloudbees/api-platform/queries/useOrganizationQueries'
import { useGetSingleTeamWithUserData } from '@cloudbees/api-platform/queries/useTeamsQueries'
import { useTeamsQuery } from '@cloudbees/api-platform/queries/useTeamsQuery'
import { useAuthenticatedUser } from '@cloudbees/shared/user'
import {
  useOrganizationContext,
  useOrganizationId,
} from '@cloudbees/shared-auth/RequireOrganization'
import { useAppContext } from '@cloudbees/shared-providers/components/AppContextProvider'
import { type IPlatformOrganization } from '@cloudbees/shared-types'
import { type IRbacAuthorizationData } from '@cloudbees/shared-types/types/rbac'
 
enum ResourceType {
  Organization = 'Organization',
  Component = 'Component',
}
 
export interface IRole {
  id?: string // The authorization ID of the role
  resourceId: string
  resourceType?: ResourceType
  resourceName: string
  roleName?: 'User' | 'Admin' | 'Owner'
}
 
export interface IPrincipalRole {
  type: 'user' | 'team'
  id: string // ID of the user or team
  name: string // Full name of the user or team name
  principal: string
  roles: IRole[]
}
 
const getAllOrgs = (org: IPlatformOrganization): IPlatformOrganization[] => {
  const orgs = [org]
 
  if (org.childOrganizations) {
    org.childOrganizations.forEach(childOrg => {
      orgs.push(...getAllOrgs(childOrg))
    })
  }
 
  return orgs
}
 
/**
 * Returns all authorization data for users and teams in the organization.
 *
 * Authorization are granted for a resource ID, so we also need to provide a map of resource IDs to resource names.
 *
 * @returns {IPrincipalRole[]} principalRoles - All users and teams in the organization with their roles
 * @returns {Record<string, string>} componentIdToNameMap - Map of component IDs to component names
 * @returns {Record<string, string>} orgIdToNameMap - Map of organization IDs to organization names
 * @returns {boolean} isLoading - Whether or not the data is still loading
 */
export const usePrincipalRoles = () => {
  const { tenantOrg } = useOrganizationContext()
  const isManagementConsole = useAppContext() === 'management-console'
  const orgId = useOrganizationId()
  const tenantOrgId = tenantOrg?.id || orgId
 
  const { data: teams } = useTeamsQuery(tenantOrgId)
  const { data: authorizationsResponse } = useAuthorizationsQuery()
  const defaultTeam = teams?.teams.find(team => team.isDefault)
  const { data: userResponse, isLoading: usersLoading } = useGetSingleTeamWithUserData(
    defaultTeam?.id,
  )
 
  const users = userResponse?.team?.users ?? []
 
  const { data: servicesResponse } = useComponentsQuery(orgId || '')
 
  const { organization: nguiOrgData } = useAuthenticatedUser()
  const { data: adminToolsOrgData } = useGetOrganization(orgId)
 
  const orgsWithSubOrgs = isManagementConsole ? nguiOrgData : adminToolsOrgData?.organization
 
  if (
    !users ||
    !authorizationsResponse ||
    !teams ||
    usersLoading ||
    !servicesResponse ||
    !orgsWithSubOrgs
  ) {
    return {
      principalRoles: [],
      componentIdToNameMap: {},
      orgIdToNameMap: {},
      isLoading: true,
    }
  }
 
  const componentIdToNameMap = servicesResponse.service.reduce<Record<string, string>>(
    (acc, service) => {
      acc[service.id] = service.name
      return acc
    },
    {},
  )
 
  const allOrgs = orgsWithSubOrgs ? getAllOrgs(orgsWithSubOrgs) : undefined
 
  const orgIdToNameMap =
    allOrgs?.reduce<Record<string, string>>(
      (acc, org) => {
        acc[org.id] = org.displayName
        return acc
      },
      {
        [orgsWithSubOrgs?.id || '']: orgsWithSubOrgs?.displayName || '',
      },
    ) || {}
 
  const getRoles = (authorizationFilter: (authorization: IRbacAuthorizationData) => boolean) =>
    authorizationsResponse.authorizations
      .filter(authorizationFilter)
      .map(({ resourceId, roleName, id }) => {
        const componentName = componentIdToNameMap[resourceId]
        const orgName = orgIdToNameMap[resourceId]
 
        if (componentName) {
          return {
            id,
            resourceType: ResourceType.Component,
            resourceId,
            resourceName: componentName,
            roleName,
          }
        }
 
        if (orgName) {
          return {
            id,
            resourceType: ResourceType.Organization,
            resourceId,
            resourceName: orgName,
            roleName,
          }
        }

        return undefined
      })
      .sort((a, b) => {
        if (a === undefined || b === undefined) {
          return 1
        }
 
        if (a.resourceType === ResourceType.Organization) {
          return -1
        }

        if (b.resourceType === ResourceType.Organization) {
          return 1
        }

        return a.resourceName.localeCompare(b.resourceName)
      })
      .filter(Boolean) as IRole[]
 
  const userRoles: IPrincipalRole[] = users.map(user => {
    const principal = user?.logins[0]?.email || ''
    const name = `${user?.givenname} ${user?.familyname}`
 
    const roles = getRoles(authorization => authorization.userId === user?.id)
 
    return {
      id: user?.id || '',
      type: 'user',
      principal,
      name,
      roles,
    }
  })
 
  const teamRoles: IPrincipalRole[] = teams.teams
    // Filter out the default team
    .filter(team => !team.isDefault)
    .map(team => {
      const principal = team.name
      const name = team.name

      const roles = getRoles(authorization => authorization.teamId === team.id)

      return {
        id: team.id,
        type: 'team',
        principal,
        name,
        roles,
      }
    })
 
  return {
    principalRoles: [...userRoles, ...teamRoles].filter(
      principalRole => ![''].includes(principalRole.principal),
    ),
    componentIdToNameMap,
    orgIdToNameMap,
    isLoading: false,
  }
}

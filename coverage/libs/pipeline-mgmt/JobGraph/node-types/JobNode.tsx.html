import { useState } from 'react'
import { useTranslation } from 'react-i18next'
import { type NodeProps, type ReactFlowState, Handle, useReactFlow, useStore } from 'reactflow'
import Box from '@mui/material/Box'
import { useTheme } from '@mui/material/styles'
 
import { Overflow } from '@cloudbees/shared-components/OverflowMenu'
import { TruncableTypography } from '@cloudbees/shared-components/TruncableTypography'
import { useColorMode } from '@cloudbees/shared-theme/UIProvider'
 
import { useGraphActions } from '../../hooks'
import { AddStepButton } from '../AddStepButton/AddStepButton'
import { NODE_HEIGHT, STEP_HEIGHT, ZOOM_BREAKPOINT } from '../constants'
import { JobSteps } from '../JobSteps/JobSteps'
import { MoreStepsButton } from '../MoreStepsButton/MoreStepsButton'
import { type IJobNode } from '../types'
 
import handledark from './icons/handledark.svg'
import handlelight from './icons/handlelight.svg'
import { useNodeStyles } from './useNodeStyles'
 
export const JobNode = (node: NodeProps<IJobNode>) => {
  const { t } = useTranslation('pipeline-management', { keyPrefix: 'job-graph' })
  const [displayedSteps, setDisplayedSteps] = useState(5)
  const { palette } = useTheme()
  const { mode } = useColorMode()
  const jobSteps = node?.data?.steps ? node?.data?.steps : []
  const { onJobClick, onDeleteJobClick, onAddStepClick } = useGraphActions()
 
  const zoomSelector = (s: ReactFlowState) => s.transform[2] >= ZOOM_BREAKPOINT
  const showSteps = useStore(zoomSelector)
  const { getEdges } = useReactFlow()
 
  const edges = getEdges()
  const triggerEdge = edges.find(edge => edge.target === node.id && edge.source === 'trigger')
 
  // height of the steps that need to be hidden
  const stepsAdjustment =
    jobSteps.length - displayedSteps > 0 ? (jobSteps.length - displayedSteps) * STEP_HEIGHT : 0
 
  const height = showSteps && node ? node.data.height - stepsAdjustment : NODE_HEIGHT
 
  const showMoreButton = stepsAdjustment > 0 && showSteps
 
  const handleStyle = {
    height: 8,
    width: 8,
    backgroundSize: 'cover',
    backgroundImage: mode === 'light' ? `url("${handlelight}")` : `url("${handledark}")`,
    border: '0px',
  }
 
  return (
    <Box height={height} sx={{ transition: 'height 500ms ease-in-out' }}>
      {node.targetPosition && (
        <Handle
          type="target"
          isConnectable={false}
          position={node.targetPosition}
          style={{
            ...handleStyle,
            opacity: triggerEdge?.hidden ? 0 : 1,
          }}
        />
      )}
      <Box
        sx={{
          ...useNodeStyles(palette),
        }}
      >
        <Box
          sx={{
            display: 'flex',
          }}
          onClick={() => onJobClick(node.data, 'UPDATE')}
        >
          <Box
            sx={{
              display: 'flex',
              width: '90px',
              alignItems: 'center',
              color: palette.text.primary,
            }}
          >
            <TruncableTypography
              tooltipTitle={node.data.label || ''}
              typographyProps={{
                sx: {
                  pl: '12px',
                  fontWeight: 400,
                  fontSize: '12px',
                },
              }}
            >
              {node.data.label || ''}
            </TruncableTypography>
          </Box>
        </Box>
        <Box
          sx={{
            display: 'flex',
            justifyContent: 'flex-end',
            alignItems: 'center',
          }}
        >
          <Overflow.Container>
            <Overflow.Button
              data-testid={`actions-button-job-${node.id}`}
              sx={{
                'padding': '0 4px 0 0',
                '&:hover': {
                  backgroundColor: 'unset',
                },
                '& .MuiSvgIcon-root': {
                  height: '18px',
                  width: 'auto',
                  color: palette.text.primary,
                  opacity: '0.6',
                },
              }}
            />
            <Overflow.Menu>
              <Overflow.MenuItem
                data-testid="edit-menu-item"
                onClick={() => onJobClick(node.data, 'UPDATE')}
              >
                {t('edit')}
              </Overflow.MenuItem>
              <Overflow.MenuItem
                data-testid="delete-menu-item"
                onClick={() => onDeleteJobClick(node.data)}
              >
                {t('delete')}
              </Overflow.MenuItem>
            </Overflow.Menu>
          </Overflow.Container>
        </Box>
      </Box>
      {!!jobSteps.length && (
        <JobSteps height={height} job={node.data} steps={jobSteps} isOpen={showSteps} />
      )}
      {showSteps && <AddStepButton onClick={() => onAddStepClick(node.data)} />}
 
      {showMoreButton && <MoreStepsButton onClick={() => setDisplayedSteps(displayedSteps + 3)} />}
      {node.sourcePosition && (
        <Handle
          type="source"
          isConnectable={false}
          position={node.sourcePosition}
          style={{ ...handleStyle, opacity: node.data.hasDownStream ? 1 : 0 }}
        />
      )}
    </Box>
  )
}

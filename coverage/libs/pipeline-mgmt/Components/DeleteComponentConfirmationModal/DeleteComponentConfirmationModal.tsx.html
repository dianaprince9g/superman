import { Trans, useTranslation } from 'react-i18next'

import { type IComponentAutomation } from '@cloudbees/api-platform/queries/guiderails.types'
import { ConfirmationModal } from '@cloudbees/shared-components/ConfirmationModal'

export interface DeleteComponentConfirmationModalProps {
  componentToDelete?: IComponentAutomation
  isOpen: boolean
  handleClose: () => void
  handleDelete: () => void
}

export const DeleteComponentConfirmationModal = (props: DeleteComponentConfirmationModalProps) => {
  const { componentToDelete, isOpen, handleClose, handleDelete } = props
  const { t } = useTranslation('pipeline-management', { keyPrefix: 'components-list' })

  return (
    <ConfirmationModal.Modal
      onClose={handleClose}
      open={isOpen}
      fullWidth={true}
      maxWidth="sm"
      data-testid="delete-component-modal"
    >
      <ConfirmationModal.Title id="customized-dialog-title">
        {t(`delete_modal.delete_title`)}
      </ConfirmationModal.Title>
      <ConfirmationModal.Content>
        <ConfirmationModal.ContentText>
          <Trans
            t={t}
            i18nKey={`delete_modal.delete_description`}
            values={{ propName: componentToDelete?.name }}
          />
        </ConfirmationModal.ContentText>
      </ConfirmationModal.Content>
      <ConfirmationModal.Actions>
        <ConfirmationModal.ActionItem data-testid="delete-cancel" onClick={handleClose}>
          {t('delete_modal.delete_cancel')}
        </ConfirmationModal.ActionItem>
        <ConfirmationModal.ActionItem
          data-testid="delete-confirmation"
          autoFocus
          color="error"
          onClick={handleDelete}
        >
          {t('delete_modal.delete_confirmation')}
        </ConfirmationModal.ActionItem>
      </ConfirmationModal.Actions>
    </ConfirmationModal.Modal>
  )
}

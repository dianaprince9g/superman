import { useTranslation } from 'react-i18next'
import { generatePath } from 'react-router-dom'
import { useQueryClient } from '@tanstack/react-query'
import { object, string } from 'yup'
 
import { type IComponentSateSummary } from '@cloudbees/api-platform/queries/guiderails.types'
import { useComponentsQuery } from '@cloudbees/api-platform/queries/useComponentsQuery'
import { useCreateComponentsMutation } from '@cloudbees/api-platform/queries/useCreateComponentsMutation'
import { useInstrumentation } from '@cloudbees/api-platform/queries/useInstrumentation'
import { useRepositoriesQuery } from '@cloudbees/api-platform/queries/useRepositoriesQuery'
import { useUpdateComponentsMutation } from '@cloudbees/api-platform/queries/useUpdateComponentsMutation'
import { WORKFLOWS_PATH } from '@cloudbees/route-config-nextgen'
import { useOrganizationId } from '@cloudbees/shared-auth/RequireOrganization'
import { useNotification } from '@cloudbees/shared-hooks/useNotification'
import { useDrawerStepperContext } from '@cloudbees/shared-patterns/DrawerStepper'
import { Form } from '@cloudbees/shared-patterns/Form'
import { handleError } from '@cloudbees/shared-utils/handleError'
import { useNavigate, useParams } from '@cloudbees/shared-utils/navigation'
 
const COMPONENT_NAME_REGEX = /^[A-Za-z0-9_.-]*$/
 
interface ICreateComponentDrawerProps {
  component?: IComponentSateSummary
}
 
export const CreateComponentNameForm = ({
  component,
}: ICreateComponentDrawerProps): JSX.Element => {
  const { t } = useTranslation('pipeline-management', {
    keyPrefix: 'create-component-drawer.create-component-name-form',
  })
  const navigate = useNavigate()
 
  const organizationId = useOrganizationId()
  // Query should always be already in cache at this stage
  const { data: repositoriesResponse } = useRepositoriesQuery({
    organizationId,
  })
  const { data: componentsResponse } = useComponentsQuery(organizationId)
 
  const { orgId = null } = useParams()
 
  const validationSchema = object().shape({
    service_name: string()
      .required(t('error-messages.component-name.required') || '')
      .min(4, t('error-messages.component-name.min') || '')
      .max(254, t('error-messages.component-name.max') || '')
      .matches(COMPONENT_NAME_REGEX, t('error-messages.component-name.matches') || ''),
    service_description: string().max(254, t('error-messages.component-description.max') || ''),
  })
 
  const { nextButton, setError, nextStep, data } = useDrawerStepperContext()
  const { notify } = useNotification()
 
  const { mutate: create } = useCreateComponentsMutation(organizationId)
  const { mutate: update } = useUpdateComponentsMutation(organizationId, component?.id)
 
  const { track } = useInstrumentation()
 
  const queryClient = useQueryClient()
  const onCreateComponentSuccess = (response: any) => {
    const updatedRepositoriesResponse = {
      repository: repositoriesResponse?.repository.map(r =>
        r.url === response?.service.repositoryUrl ? { ...r, componentId: response?.service.id } : r,
      ),
    }
    // Manually update the repository used to created the service so it can't be used twice in a row
    // Doing it manually because invalidating the query is not performant enough
    queryClient.setQueryData(['repositories', organizationId], updatedRepositoriesResponse)
 
    // Manually update the service because invalidating the query is not performant enough
    queryClient.setQueryData(['services', organizationId], {
      service: [...(componentsResponse?.service || []), response.service],
    })
 
    // The properties query always returns args in the response, but this messes
    // up the table display so we'll destructure it and only use the relevantProperties
    // (this is all of the returned properties minus the args).
    const { args, ...relevantProperties } = response ?? {}
    notify(t('component-created', { componentName: relevantProperties.service.name }), 'success')
 
    let scmProvider = ''
    try {
      scmProvider = new URL(relevantProperties.service.repositoryUrl).hostname
    } catch (e) {
      console.error(e)
    }
 
    track('Component Created', {
      componentName: relevantProperties.service.name,
      componentId: relevantProperties.service.id,
      scmProvider,
    })
 
    nextStep()
    /**
     * Add small delay before redirecting to Workflows page for SDP-6024
     *
     * Revisit this to see if there is a better solution.
     * TODO {@link https://cloudbees.atlassian.net/browse/SDP-6313 Revisit delayed redirection to Automations page}.
     */
    setTimeout(() => {
      navigate({
        pathname: generatePath(WORKFLOWS_PATH, { orgId }),
        search: `componentId=${relevantProperties.service.id}`,
      })
    }, 1500)
  }
 
  const onEditComponentSuccess = (response: any) => {
    const updatedComponentsResponse = {
      service: componentsResponse?.service.map(s =>
        s.id === response?.service.id
          ? {
              ...s,
              name: response?.service.name,
              description: response?.service.description,
            }
          : s,
      ),
    }

    // Manually update the service because invalidating the query is not performant enough
    queryClient.setQueryData(['services', organizationId], updatedComponentsResponse)

    const { args, ...relevantProperties } = response ?? {}
    notify(t('component-updated', { componentName: relevantProperties.service.name }), 'success')
    nextStep()
  }
 
  const createComponent = (values: any, actions: any) => {
    const componentPayload = {
      service: {
        name: values.service_name,
        description: values.service_description,
        endpointId: data.endpoint_id || component?.endpointId,
        repositoryUrl: data.repository_url || component?.repositoryUrl,
      },
    }
 
    if (component) {
      update(componentPayload, {
        onSuccess: onEditComponentSuccess,
        onError: (error: any) => {
          let errorMessage = t('error-messages.server-error-edit', {
            componentName: values.service_name,
          })
          if (error?.status === 409) {
            errorMessage = t('error-messages.already-created', {
              componentName: values.service_name,
            })
          }

          actions.setSubmitting(false)
          setError(errorMessage)
        },
      })
    } else {
      create(componentPayload, {
        onSuccess: onCreateComponentSuccess,
        onError: (error: any) => {
          handleError({
            error,
            actions,
            t,
            values,
            callback: errorMessage => setError(errorMessage),
          })
        },
      })
    }
  }
 
  const initialData = {
    service_name:
      component?.name ||
      data.repository_url
        ?.substring(data.repository_url?.lastIndexOf('/') + 1)
        .replace(/.git/i, ''),
    service_description: component?.description || '',
  }
 
  return (
    <Form.Context
      initialValues={initialData}
      validationSchema={validationSchema}
      onSubmit={createComponent}
      validateOnMount={true}
    >
      <Form.Form>
        <Form.TextField
          fullWidth
          id={'service_name'}
          name={'service_name'}
          label={t('component-name-label')}
          inputProps={{ 'data-testid': 'service-name' }}
          placeholder={t('component-name-placeholder') || ''}
          sx={{ mb: 3 }}
        />
        <Form.TextField
          fullWidth
          multiline
          minRows={6}
          id={'service_description'}
          name={'service_description'}
          label={t('component-description-label')}
          inputProps={{ 'data-testid': 'service-description' }}
          placeholder={t('component-description-placeholder') || ''}
        />
        {nextButton(
          // We have to set dirty attribute  as true when we add a new component.
          // In this case user already has a  default component name that equals corresponding repository name
          // and can create component with this default nam
          <Form.Submit dirty={!component?.name}>{component ? t('save') : t('create')}</Form.Submit>,
        )}
      </Form.Form>
    </Form.Context>
  )
}

import { useTranslation } from 'react-i18next'
import Button from '@mui/material/Button'
import DOMPurify from 'dompurify'
import { v4 as uuid } from 'uuid'
 
import { Action, useInstrumentation } from '@cloudbees/api-platform/queries'
import { Drawer } from '@cloudbees/shared-patterns/Drawer'
import { Form } from '@cloudbees/shared-patterns/Form'
 
import { useYaml } from '../../context/YamlProvider'
import { type ITriggerInput, type ITriggerInputs, ParamType } from '../../DSL/dsl.types'
 
import { CreateTriggerForm } from './CreateTriggerForm'
import {
  type ChoiceOption,
  type ICreateTriggerDrawerProps,
  type ITriggerFormValues,
} from './Trigger.types'
import { getValidationSchema } from './validationSchema'
 
export const CreateTriggerDrawer = ({
  open,
  onClose,
  triggerTypes,
  branches,
  selectedTrigger,
  editMode,
}: ICreateTriggerDrawerProps) => {
  const { t } = useTranslation('pipeline-management', {
    keyPrefix: 'trigger-drawer',
  })
 
  const { createTrigger } = useYaml()
  const { track } = useInstrumentation()
 
  const initialValues: ITriggerFormValues = {
    type: selectedTrigger?.type || '',
    branches: selectedTrigger?.branches || [],
    inputs: Object.entries(selectedTrigger?.inputs ?? {}).map(([key, parameter]) => {
      const options = parameter?.options?.map(option => ({
        option: option,
        id: uuid(),
      }))

      return {
        ...parameter,
        name: key,
        id: uuid(),
        options,
        default:
          parameter.type === ParamType.CHOICE
            ? ({
                option: parameter.default,
                id: options?.find(i => i.option === parameter.default)?.id ?? uuid(),
              } as ChoiceOption)
            : parameter.default,
        required: parameter.required === true,
      }
    }),
  }
 
  const onSubmit = (values: ITriggerFormValues) => {
    // Change the trigger type if the form has a different type than the original value
    const replacedType =
      selectedTrigger && selectedTrigger.type !== values.type ? selectedTrigger.type : undefined

    let inputs: ITriggerInputs = {}
    if (values?.inputs) {
      inputs = values.inputs.reduce((acc, obj) => {
        const parameter: ITriggerInput = {
          type: obj.type,
        }
        if (obj.type === 'choice') {
          parameter.options = obj.options?.map(option => DOMPurify.sanitize(option.option))
          parameter.default = obj.default
            ? DOMPurify.sanitize((obj.default as ChoiceOption).option)
            : undefined
        }
        if (obj.default && obj.type !== 'choice') {
          parameter.default =
            typeof obj.default === 'string' ? DOMPurify.sanitize(obj.default) : obj.default
        }
        if (obj.type === 'boolean') {
          parameter.default = obj.default ? true : false
        }
        if (obj.required) {
          parameter.required = obj.required
        }
        if (obj.description) {
          parameter.description = obj.description
        }
        return {
          ...acc,
          [obj.name]: parameter,
        }
      }, {})
    }

    createTrigger(values.type, values.branches, inputs, replacedType)

    console.log({ selectedTrigger, type: values.type })
    track('Automation Graphical Composer Used', {
      type: 'TRIGGER',
      action: selectedTrigger ? Action.UPDATE : Action.CREATE,
    })

    onClose()
  }
 
  return (
    <Drawer.Wrapper open={open} onClose={onClose}>
      <Form.Context<ITriggerFormValues>
        initialValues={initialValues}
        onSubmit={onSubmit}
        validationSchema={getValidationSchema(t)}
      >
        <>
          <Drawer.Header title={editMode === 'UPDATE' ? t('edit-title') : t('create-title')} />
          <Drawer.Content>
            <CreateTriggerForm
              triggerTypes={triggerTypes}
              selectedTrigger={selectedTrigger}
              branches={branches}
            />
          </Drawer.Content>
          <Drawer.Footer>
            <Button variant="outlined" onClick={() => onClose()}>
              {t('cancel')}
            </Button>
            <Form.Submit>{t('save')}</Form.Submit>
          </Drawer.Footer>
        </>
      </Form.Context>
    </Drawer.Wrapper>
  )
}

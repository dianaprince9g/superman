import { useState } from 'react'
import { useTranslation } from 'react-i18next'
import Button from '@mui/material/Button'
import CircularProgress from '@mui/material/CircularProgress'
import Typography from '@mui/material/Typography'
 
import { type IComponentSateSummary, useMoveComponentQuery } from '@cloudbees/api-platform/queries'
import { useOrganizationContext } from '@cloudbees/shared-auth/RequireOrganization'
import { useNotification } from '@cloudbees/shared-hooks/useNotification'
import { useDrawerStepperContext } from '@cloudbees/shared-patterns/DrawerStepper'
import { type IPlatformOrganization } from '@cloudbees/shared-types'
import { OrganizationSelector } from '@cloudbees/user-management/OrganizationsPage/components/OrganizationSelector'
 
interface SelectOrganizationStepProps {
  component: IComponentSateSummary
}
 
export const SelectOrganizationStep = ({ component }: SelectOrganizationStepProps) => {
  const { t } = useTranslation('pipeline-management', { keyPrefix: 'move-component-drawer' })
  const { nextButton, nextStep } = useDrawerStepperContext()
  const { notify } = useNotification()
  const organization = useOrganizationContext()
  const [selectedOrganization, setSelectedOrganization] = useState<IPlatformOrganization>()
 
  const { mutate, isLoading: isSubmitting } = useMoveComponentQuery({
    componentId: component.id,
    organizationId: organization.id,
  })
 
  const moveComponent = () => {
    mutate(
      {
        resource: {
          id: component.id,
          parentId: selectedOrganization?.id,
          name: component.name,
          type: 'RESOURCE_TYPE_REPOSITORY',
        },
      },
      {
        onSuccess: () => {
          notify(
            t('move-success', {
              componentName: component.name,
              destinationOrganizationName:
                selectedOrganization?.displayName ?? selectedOrganization?.domainName,
            }),
            'success',
          )
        },
        onError: async (e: any) => {
          const error = await e.json()
          notify(`${error.message}`, 'error')
        },
        onSettled: () => nextStep(),
      },
    )
  }
 
  return (
    <>
      <Typography sx={{ mb: 2 }}>{t('select-organization')}</Typography>
      <OrganizationSelector
        onChange={org => {
          setSelectedOrganization(org)
        }}
      />
      {nextButton(
        <Button
          onClick={moveComponent}
          variant="contained"
          disabled={!selectedOrganization?.id || isSubmitting}
        >
          {isSubmitting ? <CircularProgress color="inherit" size={24} /> : t('move')}
        </Button>,
      )}
    </>
  )
}

import { useState } from 'react'
import { useTranslation } from 'react-i18next'
import VerticalSplitOutlinedIcon from '@mui/icons-material/VerticalSplitOutlined'
import ViewStreamOutlinedIcon from '@mui/icons-material/ViewStreamOutlined'
import Autocomplete from '@mui/material/Autocomplete'
import Box from '@mui/material/Box'
import Button from '@mui/material/Button'
import FormControl from '@mui/material/FormControl'
import IconButton from '@mui/material/IconButton'
import Stack from '@mui/material/Stack'
import TextField from '@mui/material/TextField'
import ToggleButton from '@mui/material/ToggleButton'
import ToggleButtonGroup from '@mui/material/ToggleButtonGroup'
import Tooltip from '@mui/material/Tooltip'
 
import { useOrganizationPermission } from '@cloudbees/api-platform/Permission'
import { RBAC_TYPES } from '@cloudbees/shared-types'
import { useParams } from '@cloudbees/shared-utils/navigation'
 
import { useYaml } from '../../context/YamlProvider'
import { ButtonOriginType, ManualTrigger } from '../../ManualTrigger'
import { useSplitViewContext } from '../SplitView/SplitViewProvider'
 
type ViewMode = 'split' | 'visual' | 'script'
 
interface IEditAutomationMenuProps {
  selectedBranch: string
  setSelectedBranch: (v: string) => void
  onCommit: () => void
  onCancel: () => void
  branches: string[]
  modified: boolean
}
 
export const EditWorkflowMenu = ({
  selectedBranch,
  setSelectedBranch,
  onCommit,
  onCancel,
  branches,
  modified,
}: IEditAutomationMenuProps) => {
  const { t } = useTranslation('pipeline-management', {
    keyPrefix: 'automations.menu',
  })
 
  const [viewMode, setViewMode] = useState<ViewMode>('split')
  const { parsingErrors, validationErrors } = useYaml()
  const { componentId = '', workflowId = '' } = useParams()
  const { canUpdate } = useOrganizationPermission(RBAC_TYPES.SERVICE)
  const { setMode, mode, lastSplitView } = useSplitViewContext()
 
  const handleOrientationChange = () => {
    if (mode === 'SPLIT_HORIZONTAL') {
      setMode('SPLIT_VERTICAL')
    } else if (mode === 'SPLIT_VERTICAL') {
      setMode('SPLIT_HORIZONTAL')
    }
  }
 
  const handleViewModeChange = (event: React.MouseEvent<HTMLElement>, newViewMode: string) => {
    if (!newViewMode) {
      return
    }
    setViewMode(newViewMode as ViewMode)
    if (newViewMode === 'split') {
      setMode(lastSplitView)
    } else if (newViewMode === 'script') {
      setMode('SCRIPT_ONLY')
    } else {
      setMode('VISUAL_ONLY')
    }
  }
 
  const orientationToggleDisabled = !['SPLIT_HORIZONTAL', 'SPLIT_VERTICAL'].includes(mode)
 
  return (
    <Stack
      direction={'row'}
      justifyContent={'space-between'}
      alignItems={'center'}
      width={'100%'}
      px={4}
      mt={2}
    >
      <Box width={500} display="flex" alignItems="center" gap={2}>
        <FormControl fullWidth>
          <Autocomplete
            disablePortal
            id="select-automation-branch"
            autoHighlight
            defaultValue={selectedBranch}
            ListboxProps={{ style: { maxHeight: 400 } }}
            options={branches}
            isOptionEqualToValue={(o, v) => o === v}
            onChange={(e, value) => setSelectedBranch(value || '')}
            value={selectedBranch || ''}
            clearIcon={null}
            size="small"
            renderInput={params => (
              <TextField
                {...params}
                inputProps={{ ...params.inputProps }}
                id="branches-textfield"
                name="branches"
                label={t('branch')}
              />
            )}
          />
        </FormControl>
        <Button
          variant={'text'}
          disabled={!modified}
          onClick={onCancel}
          sx={{ marginLeft: 5, marginRight: 1 }}
        >
          {t('cancel')}
        </Button>
        <Tooltip title={!canUpdate ? t('tooltip_edit_permission') : ''} placement="top">
          <span>
            <Button
              data-testid="commit-button"
              variant="contained"
              onClick={onCommit}
              disabled={
                !canUpdate ||
                !modified ||
                Boolean(parsingErrors.length) ||
                Boolean(validationErrors.length)
              }
            >
              {t('commit-changes')}
            </Button>
          </span>
        </Tooltip>
 
        <ManualTrigger
          isYamlModified={modified}
          workflowId={workflowId}
          componentId={componentId}
          variant="button"
          buttonOrigin={ButtonOriginType.WorkflowEditRun}
          trackingEvent="Workflow Edit Run"
        />
      </Box>
      <Box sx={{ marginLeft: '1rem', display: 'flex' }}>
        <Tooltip title={t('tooltip_orientation_toggle')} placement="top">
          <span>
            <IconButton
              onClick={handleOrientationChange}
              disabled={orientationToggleDisabled}
              data-testid="orientation-toggle"
            >
              {mode === 'SPLIT_HORIZONTAL' ? (
                <VerticalSplitOutlinedIcon />
              ) : (
                <ViewStreamOutlinedIcon />
              )}
            </IconButton>
          </span>
        </Tooltip>
 
        <ToggleButtonGroup
          value={viewMode}
          onChange={handleViewModeChange}
          color="primary"
          size="small"
          exclusive
        >
          <ToggleButton data-testid="split-button" value="split">
            {t('split')}
          </ToggleButton>
          <ToggleButton data-testid="script-button" value="script">
            {t('script')}
          </ToggleButton>
          <ToggleButton data-testid="visual-button" value="visual">
            {t('visual')}
          </ToggleButton>
        </ToggleButtonGroup>
      </Box>
    </Stack>
  )
}

import { useState } from 'react'
import { useTranslation } from 'react-i18next'
import ExpandMoreIcon from '@mui/icons-material/ExpandMore'
import Accordion from '@mui/material/Accordion'
import AccordionDetails from '@mui/material/AccordionDetails'
import AccordionSummary from '@mui/material/AccordionSummary'
import Menu from '@mui/material/Menu'
import Stack from '@mui/material/Stack'

import { useComponentsQuery } from '@cloudbees/api-platform/queries/useComponentsQuery'
import { useResourceListQuery } from '@cloudbees/api-platform/queries/useResourceListQuery'
import { useAuthenticatedUser } from '@cloudbees/shared/user'
import { Filter } from '@cloudbees/shared-patterns/Filter'
import { useParams } from '@cloudbees/shared-utils/navigation'

interface IRunsTableFilterAccordionProps {
  title: string
  attribute: string
  multiSelect?: boolean
}

const RunsTableFilterAccordion = ({
  title,
  attribute,
  multiSelect,
}: IRunsTableFilterAccordionProps) => {
  return (
    <Accordion disableGutters data-testid={`${attribute}-accordion`}>
      <AccordionSummary
        sx={{
          px: 1,
          flexDirection: 'row-reverse',
          gap: 1,
        }}
        expandIcon={<ExpandMoreIcon />}
      >
        {title}
      </AccordionSummary>
      <AccordionDetails>
        {multiSelect ? (
          <Filter.MultiSelect attribute={attribute} />
        ) : (
          <Filter.Select placeholder={title} attribute={attribute} />
        )}
      </AccordionDetails>
    </Accordion>
  )
}

export const RunsTableInfiniteFilter = () => {
  const { t } = useTranslation('pipeline-management', {
    keyPrefix: 'runs-page.filters',
  })

  const { organization } = useAuthenticatedUser()

  const { componentId } = useParams()

  const [anchorEl, setAnchorEl] = useState<null | HTMLElement>(null)

  const { data: branches = [] } = useResourceListQuery('RESOURCE_TYPE_BRANCH')
  const { data: pullRequests = [] } = useResourceListQuery('RESOURCE_TYPE_PULL_REQUEST')
  const { data: workflows } = useResourceListQuery('RESOURCE_TYPE_AUTOMATION')
  const { data: componentsResponse } = useComponentsQuery(organization?.id)

  const branchesAndPRs = [...branches, ...pullRequests].sort((resourceA, resourceB) =>
    resourceA.localeCompare(resourceB),
  )

  const branchArray = branchesAndPRs?.map(branch => ({ branch })) || []
  const workflowsArray = workflows?.map(workflow => ({ workflow })) || []
  const componentsArray =
    componentsResponse?.service
      ?.map(component => ({
        component: component.name,
      }))
      ?.sort((componentA, componentB) =>
        componentA.component.localeCompare(componentB.component),
      ) || []

  return (
    <Filter.Container
      density="compact"
      data={[...branchArray, ...workflowsArray, ...componentsArray]}
    >
      <Stack width="560px" direction="row" gap={1} paddingBottom={2}>
        <Filter.Button handleClick={e => setAnchorEl(e.currentTarget)} />
        <Filter.Search placeholder="Search all run summaries" attributes={['']} />
      </Stack>
      <Menu
        open={Boolean(anchorEl)}
        onClose={() => setAnchorEl(null)}
        anchorEl={anchorEl}
        sx={{
          '& .MuiList-root': {
            p: 0,
            minWidth: '320px',
          },
        }}
        keepMounted
      >
        {!componentId && <RunsTableFilterAccordion title={t('component')} attribute="component" />}
        <RunsTableFilterAccordion title={t('source')} attribute="branch" />
        <RunsTableFilterAccordion multiSelect title={t('workflow')} attribute="workflow" />
      </Menu>
    </Filter.Container>
  )
}

import { memo, useMemo, useRef } from 'react'
import { useSearchParams } from 'react-router-dom'
import { ViewportList } from 'react-viewport-list'
import Box from '@mui/material/Box'
import Button from '@mui/material/Button'
import CircularProgress from '@mui/material/CircularProgress'
 
import { type ILogLine } from '@cloudbees/api-platform/queries/logs/useStepLogs'
import { Duration } from '@cloudbees/shared-components/Duration'
import { useIsInView } from '@cloudbees/shared-hooks/useIsInView'
import { type IStep } from '@cloudbees/shared-types'
 
import { CollapsibleRow } from './CollapsibleRow'
import { LogLine } from './LogLine'
import { useStepLogs } from './StepLogsProvider'
 
interface IStepProps {
  step: IStep
  jobId: string
  toggleExpanded: (stepId: string) => void
  expanded: boolean
  showTimestamps?: boolean
  viewportRef: React.RefObject<HTMLDivElement>
}
 
const emptyLine: ILogLine = {
  date: '',
  log: 'No logs available',
}
 
/**
 * This regex matches a typical log line that starts with a number followed by a newline and the log content.
 */
const LOG_LINE_REGEX = /(\d+)\n\s*(?=\D)/g
 
const StepComponent = ({
  step,
  jobId,
  toggleExpanded,
  expanded,
  showTimestamps,
  viewportRef,
}: IStepProps) => {
  const [searchParams] = useSearchParams()
  const lineParam = searchParams.get('line')
 
  const initialIndex = lineParam ? parseInt(lineParam) || 0 : undefined
 
  const useStepProps = useMemo(
    () => ({
      jobId,
      step,
      expanded,
      expand: () => {
        if (!expanded) {
          toggleExpanded(step.id)
        }
      },
    }),
    [expanded, jobId, step, toggleExpanded],
  )
 
  const { data, isLoading, isStreaming, listRef } = useStepLogs(useStepProps)
  //NOTE - filter step data to avoid auxiliary logs without date to be shown
  const stepLogsData = data?.filter(step => step.date)
  const isStepUnknownStatus = step.status === 'UNKNOWN'
 
  const startTime = step.startedAt ? new Date(step.startedAt) : undefined
  const endTime = new Date(step.updatedAt)
 
  const overflowAnchorRef = useRef<HTMLDivElement>(null)
 
  const goToEnd = () => {
    const offset = viewportRef.current?.clientHeight || 0

    if (stepLogsData && listRef.current) {
      listRef.current.scrollToIndex({
        index: stepLogsData.length - 1,
        alignToTop: false,
        offset: offset / 2,
        prerender: 100,
      })
    }
  }
 
  const isEndVisible = useIsInView({
    containerRef: viewportRef,
    targetRef: overflowAnchorRef.current,
  })
 
  return (
    <>
      <CollapsibleRow
        expanded={!isLoading && expanded}
        loading={isLoading && expanded}
        status={step.status}
        title={step.name}
        sx={{
          zIndex: 10,
          pl: 7,
          position: expanded ? 'sticky' : undefined,
          top: expanded ? 130 : undefined,
          backgroundColor: theme =>
            theme.palette.mode === 'dark' ? '#1D1D1D' : theme.palette.grey[200],
        }}
        onExpand={
          isStepUnknownStatus
            ? undefined
            : () => {
                toggleExpanded(step.id)
              }
        }
        decoration={
          <>
            {expanded && !isEndVisible && stepLogsData && (
              <Button onClick={() => goToEnd()}>Go To End</Button>
            )}
            {!!startTime && (
              <Duration
                startTime={startTime.getTime() / 1000}
                endTime={endTime?.getTime() / 1000}
              />
            )}
          </>
        }
      >
        <Box
          /**
           * This is a workaround to avoid newlines between the log number and its content when copying the logs to the clipboard.
           */
          onCopy={event => {
            event.preventDefault()
            const text = document?.getSelection()?.toString() ?? ''
            const formattedLogText = text.replace(LOG_LINE_REGEX, '$1 ')
            event.clipboardData.setData('text/plain', formattedLogText)
          }}
          sx={{ py: expanded ? 2 : 0, overflowAnchor: 'none' }}
        >
          {expanded && !isLoading && (
            <ViewportList
              ref={listRef}
              items={stepLogsData?.length === 0 ? [emptyLine] : stepLogsData}
              viewportRef={viewportRef}
              overscan={100}
              initialIndex={initialIndex}
              initialOffset={initialIndex ? -300 : undefined}
            >
              {(logLine, index) => {
                return (
                  <Box
                    key={index}
                    sx={{
                      fontFamily: 'Monospace',
                      verflowAnchor: 'none',
                    }}
                  >
                    <LogLine
                      logLine={logLine}
                      index={index}
                      showTimestamps={showTimestamps}
                      stepId={step.id}
                    />
                  </Box>
                )
              }}
            </ViewportList>
          )}
          {isStreaming && <CircularProgress size={16} sx={{ ml: 3.5 }} />}
        </Box>
      </CollapsibleRow>
      <Box
        ref={overflowAnchorRef}
        sx={{ overflowAnchor: 'auto', height: expanded ? '1px' : '0px' }}
      />
    </>
  )
}
 
export const Step = memo(StepComponent, (prevProps, nextProps) => {
  return (
    prevProps.step.id === nextProps.step.id &&
    prevProps.step.status === nextProps.step.status &&
    prevProps.expanded === nextProps.expanded &&
    prevProps.showTimestamps === nextProps.showTimestamps
  )
})

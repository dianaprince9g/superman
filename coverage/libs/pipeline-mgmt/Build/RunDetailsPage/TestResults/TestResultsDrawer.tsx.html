import Divider from '@mui/material/Divider'
 
import { Drawer } from '@cloudbees/shared-patterns/Drawer'
 
import { useTestResultsContext } from './TestResultsProvider'
 
export const TestResultsDrawer = () => {
  const { openDrawer, setOpenDrawer, drawerInfo, currentStep, setCurrentStep } =
    useTestResultsContext()
 
  const onClose = () => {
    if (currentStep === 0) setOpenDrawer(prev => !prev)
    else setCurrentStep(prev => prev - 1)
  }
 
  const { info, element } = drawerInfo.steps[currentStep]
 
  return (
    <Drawer.Wrapper size="xl" open={openDrawer} onClose={onClose} data-testid="widget-drawer">
      <Drawer.Header title={info.reportTitle} onClose={onClose}></Drawer.Header>
      <Divider />
      <Drawer.Content>{element}</Drawer.Content>
    </Drawer.Wrapper>
  )
}

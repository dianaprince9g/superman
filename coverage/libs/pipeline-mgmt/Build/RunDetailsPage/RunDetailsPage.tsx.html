import { type SyntheticEvent, useCallback, useEffect, useMemo, useRef, useState } from 'react'
import { useTranslation } from 'react-i18next'
import {
  type ImperativePanelHandle,
  type PanelProps,
  Panel,
  PanelGroup,
  PanelResizeHandle,
} from 'react-resizable-panels'
import { useSearchParams } from 'react-router-dom'
import DragIndicator from '@mui/icons-material/DragIndicator'
import TabContext from '@mui/lab/TabContext'
import TabList from '@mui/lab/TabList'
import TabPanel from '@mui/lab/TabPanel'
import Badge, { type BadgeProps } from '@mui/material/Badge'
import Box from '@mui/material/Box'
import Container from '@mui/material/Container'
import Divider from '@mui/material/Divider'
import Tab from '@mui/material/Tab'
import parseISO from 'date-fns/parseISO'
import useResizeObserver from 'use-resize-observer'

import { useAutomationRunQuery } from '@cloudbees/api-platform/queries/useAutomationRunQuery'
import {
  type IUseRunDetailsTestResultsProps,
  useRunDetailsTestResultsQuery,
} from '@cloudbees/api-platform/queries/useRunDetailsTestResults'
import { useWorkflowArtifacts } from '@cloudbees/api-platform/queries/workflows/useWorkflowArtifacts'
import { type IStatus } from '@cloudbees/shared-types'
import {
  type ServiceAutomationRun,
  AutomationTriggerEventType,
} from '@cloudbees/shared-types/generated'
import { useParams } from '@cloudbees/shared-utils/navigation'
import { usePreferences } from '@cloudbees/user-management/Preferences/context/PreferencesContext'

import { getTriggerInfo } from '../../utils/getTriggerInfo'
import { getCronAsHuman } from '../../Workflows/WorkflowEdit/Trigger/CronScheduler/util'
import { getElementHeight } from '../utils'

import { EvidencePanel } from './evidence/EvidencePanel'
import { RunDetailsHeader } from './header/RunDetailsHeader'
import { JobsPanel } from './jobs/JobsPanel'
import { StepsPanel } from './steps/StepsPanel'
import { TestResults } from './TestResults/TestResults'
import { type ITestResultsData, TestResultsProvider } from './TestResults/TestResultsProvider'
import { InputParameterSection } from './trigger/InputParameterSection'
import { TriggerBar } from './trigger'
import { workflowTriggerTypeMap } from './utils'

const LOCAL_STORAGE_GRAPH_KEY = 'showGraph'

const TABS = {
  JOBS: 'Jobs',
  TEST_RESULTS: 'Test Results',
}

const JOBS_SUB_TABS = {
  LOGS: 'logs',
  EVIDENCE: 'evidence',
}

const ACTIVE_TAB = 'activeTab'

const ACTIVE_JOBS_SUB_TAB = 'activeJobsSubTab'

export const RunDetailsPage = () => {
  const buildTreeWrapperRef = useRef<HTMLDivElement>(null)
  const tabListRef = useRef<HTMLDivElement>(null)
  const stepWrapperRef = useRef<HTMLDivElement>(null)
  const { formatDate } = usePreferences()
  const [buildTreeHeight, setBuildTreeHeight] = useState<string>()
  const [showGraph, setShowGraph] = useState<boolean>(
    window.localStorage.getItem(LOCAL_STORAGE_GRAPH_KEY) !== 'false',
  )

  const { t } = useTranslation('pipeline-management')

  const { data } = useAutomationRunQuery()

  const { data: workflowArtifactsResponse } = useWorkflowArtifacts()
  const automationBranch = data?.automation?.target?.name || 'main'

  useResizeObserver<HTMLDivElement>({
    ref: buildTreeWrapperRef,
    onResize: () => {
      const height = getElementHeight(buildTreeWrapperRef.current as HTMLElement)

      if (buildTreeHeight !== height) {
        setBuildTreeHeight(height)
      }
    },
  })

  const [searchParams, setSearchParams] = useSearchParams()
  const selectedJobId = searchParams.get('job')

  const triggerInfo = data?.run && getTriggerInfo(data?.run as unknown as ServiceAutomationRun)
  const inputs = data?.run?.inputs
  const selectJob = useCallback(
    (jobId: string) => {
      setSearchParams(
        prev => {
          prev.delete('line')
          prev.delete('step')
          prev.set('job', jobId)

          return prev
        },
        {
          replace: true,
        },
      )
    },
    [setSearchParams],
  )

  const cronAsHuman = useMemo(() => {
    const cron = triggerInfo?.rerunInfo?.cron
    if (cron) {
      const [minute, hour, day, month, week_day] = cron.split(' ')
      return getCronAsHuman({
        localSchedule: { minute, hour, day, month, 'week day': week_day },
      })
    }
    return ''
  }, [triggerInfo])

  const triggerBarDetails = useMemo(() => {
    if (data && data?.isComplianceWorkflow) {
      return {
        title: t(`job-graph.triggers.push`),
        description: `${
          workflowTriggerTypeMap(t)[triggerInfo?.trigger.triggerType as AutomationTriggerEventType]
            ?.description
        } ${t('runs-page.table.row.implicit-triggered-by')}`,
        icon: workflowTriggerTypeMap(t)['PUSH']?.icon,
      }
    } else if (triggerInfo?.trigger.triggerType === AutomationTriggerEventType.SCHEDULE) {
      return {
        title: `${t('pipeline-information.scheduled')} ${cronAsHuman}`,
        description: '',
        icon: workflowTriggerTypeMap(t)[
          triggerInfo?.trigger.triggerType as AutomationTriggerEventType
        ]?.icon,
      }
    }
    return {
      title: t(`job-graph.triggers.${triggerInfo?.trigger.triggerType?.toLowerCase()}`),
      description: `${
        workflowTriggerTypeMap(t)[triggerInfo?.trigger.triggerType as AutomationTriggerEventType]
          ?.description
      } ${triggerInfo?.triggerDetails?.triggeredBy.displayname}`,
      icon: workflowTriggerTypeMap(t)[
        triggerInfo?.trigger.triggerType as AutomationTriggerEventType
      ]?.icon,
    }
  }, [data?.isComplianceWorkflow])

  const runStatus: IStatus = data?.run?.status ?? 'UNKNOWN'

  useEffect(() => {
    if (data && !selectedJobId) {
      const focusedJob =
        runStatus === 'SUCCEEDED'
          ? data?.jobs[data?.jobs.length - 1]
          : data?.jobs.find(job => ['FAILED', 'STARTED', 'ABORTED'].includes(job.status))
      if (focusedJob) {
        selectJob(focusedJob.id)
      }
    }
  }, [data, selectJob, selectedJobId])

  const startTime = data?.run?.startedAt
  const startDate = startTime ? parseISO(startTime) : undefined
  const endTime = data?.run?.updatedAt
  const endDate =
    endTime && ['FAILED', 'SUCCEEDED', 'ERRORED', 'ABORTED', 'TIMED_OUT'].includes(runStatus)
      ? parseISO(endTime)
      : undefined

  const leftPanelSize: PanelProps = {
    defaultSize: showGraph ? 35 : 15,
    minSize: 15,
    maxSize: 80,
  }

  const rightPanelSize: PanelProps = {
    defaultSize: showGraph ? 65 : 85,
    minSize: 20,
    maxSize: 85,
  }

  const leftPanelRef = useRef<ImperativePanelHandle>(null)

  const handleTabChange = useCallback(
    (_: SyntheticEvent, newTab: string, tabName: string) => {
      setSearchParams(
        prev => {
          prev.set(tabName, newTab)
          return prev
        },
        {
          replace: true,
        },
      )
    },
    [setSearchParams],
  )

  const activeTab = searchParams.get(ACTIVE_TAB) || TABS.JOBS

  const activeJobsSubTab = searchParams.get(ACTIVE_JOBS_SUB_TAB) || JOBS_SUB_TABS.LOGS

  const tabListHeight = `calc(100% - ${tabListRef.current?.clientHeight ?? 0}px)`

  const { componentId: component_id, correlationId: run_id } = useParams()
  const { timezone: time_zone } = usePreferences()
  const { isLoading: testResultsLoading, data: testResultsResponse } =
    useRunDetailsTestResultsQuery({
      view_option: '',
      report_id: 'run-details-test-results-indicators',
      report_info: {
        run_id: run_id,
        component_id: component_id,
      },
      time_zone: time_zone,
    } as IUseRunDetailsTestResultsProps)

  const testResultsData = useMemo(() => {
    const testResultIndicator = (testResultsResponse?.reports?.find(report => report) ||
      {}) as ITestResultsData

    const { testCasesFailed, testCasesSkipped, isTestInsightsDataFound } = testResultIndicator

    let badgeColor: BadgeProps['color'] = 'default'
    if (testCasesFailed > 0) {
      badgeColor = 'error'
    } else if (testCasesSkipped > 0) {
      badgeColor = 'warning'
    }

    return { isTestInsightsDataFound, badgeColor }
  }, [testResultsResponse?.reports])

  return (
    <Container
      sx={theme => ({
        height: '100%',
        display: 'flex',
        flexDirection: 'column',
        [theme.breakpoints.down('md')]: {
          height: 'auto',
          pl: 0,
        },
      })}
      maxWidth={false}
      disableGutters
    >
      <RunDetailsHeader
        startDate={startDate}
        endDate={endDate}
        status={runStatus}
        branch={automationBranch}
        pendingApprovalJob={data?.jobs.find(job => job.status === 'PENDING_APPROVAL')}
        run={data?.run}
      />

      <TabContext value={activeTab}>
        <TabList
          onChange={(event, val) => handleTabChange(event, val, ACTIVE_TAB)}
          sx={theme => ({ borderBottom: `1px solid ${theme.palette.divider}` })}
          ref={tabListRef}
        >
          <Tab label={t('run-details-tabs.jobs')} value={TABS.JOBS} sx={{ ml: 2 }} />
          <Tab
            label={
              <Badge
                color={testResultsData?.badgeColor}
                anchorOrigin={{
                  vertical: 'top',
                  horizontal: 'right',
                }}
                sx={{
                  '& .MuiBadge-badge': {
                    width: '10px',
                    height: '10px',
                    borderRadius: '50%',
                    left: '100%',
                  },
                }}
                variant="dot"
              >
                {t('run-details-tabs.test-results')}
              </Badge>
            }
            value={TABS.TEST_RESULTS}
          />
        </TabList>
        <TabPanel value={TABS.JOBS} sx={{ p: 0, height: tabListHeight }}>
          <Box sx={{ p: 2 }}>
            <PanelGroup direction="horizontal">
              <Panel {...leftPanelSize} ref={leftPanelRef}>
                <Box
                  ref={buildTreeWrapperRef}
                  sx={theme => ({
                    overflowY: 'auto',
                    height: buildTreeHeight,
                    border: 1,
                    borderColor: theme.customComponents.RunDetailsPage.borderColor,
                    borderRadius: '4px',
                    backgroundColor: theme =>
                      theme.palette.mode === 'dark' ? '#1D1D1D' : theme.palette.grey[200],
                  })}
                >
                  <JobsPanel
                    jobs={data?.jobs}
                    selectedJobId={selectedJobId}
                    onSelectJob={selectJob}
                    showGraph={showGraph}
                    triggerIcon={
                      triggerInfo?.trigger.triggerType
                        ? workflowTriggerTypeMap(t)[
                            triggerInfo?.trigger.triggerType as AutomationTriggerEventType
                          ].icon
                        : undefined
                    }
                    onCollapsePanelClick={() => {
                      setShowGraph(prevShowGraph => {
                        leftPanelRef.current?.resize(!prevShowGraph ? 35 : 15)
                        window.localStorage.setItem(
                          LOCAL_STORAGE_GRAPH_KEY,
                          (!prevShowGraph).toString(),
                        )
                        return !prevShowGraph
                      })
                    }}
                  />
                </Box>
              </Panel>
              {showGraph ? (
                <PanelResizeHandle>
                  <Divider orientation="vertical" sx={{ px: 0.5 }}>
                    <DragIndicator />
                  </Divider>
                </PanelResizeHandle>
              ) : (
                <Box sx={{ px: 2 }}></Box>
              )}
              <Panel {...rightPanelSize}>
                <Box
                  sx={theme => ({
                    border: 1,
                    borderColor: theme.customComponents.RunDetailsPage.borderColor,
                    borderRadius: '4px',
                    overflowY: 'hidden',
                    height: buildTreeHeight,
                  })}
                >
                  <TabContext value={activeJobsSubTab}>
                    <TabList
                      onChange={(event, val) => handleTabChange(event, val, ACTIVE_JOBS_SUB_TAB)}
                      sx={theme => ({
                        backgroundColor: theme =>
                          theme.palette.mode === 'dark'
                            ? theme.palette.background.paper
                            : theme.palette.common.white,
                      })}
                      ref={tabListRef}
                    >
                      <Tab
                        label={t('run-details-job-sub-tabs.logs')}
                        value={JOBS_SUB_TABS.LOGS}
                        sx={{ ml: '24px' }}
                      />
                      <Tab
                        label={t('run-details-job-sub-tabs.evidence')}
                        value={JOBS_SUB_TABS.EVIDENCE}
                      />
                    </TabList>
                    <TabPanel
                      ref={stepWrapperRef}
                      value={JOBS_SUB_TABS.LOGS}
                      sx={{ p: 0, height: tabListHeight, overflow: 'auto' }}
                    >
                      {selectedJobId === 'trigger' ? (
                        <>
                          <TriggerBar
                            data-testid="run-details-trigger-bar"
                            icon={triggerBarDetails.icon}
                            title={triggerBarDetails.title}
                            branch={automationBranch}
                            description={triggerBarDetails.description}
                            startTime={formatDate(data?.run.startedAt ?? '', 'time')}
                          />
                          {Boolean(inputs?.length) && <InputParameterSection data={inputs ?? []} />}
                        </>
                      ) : (
                        <StepsPanel
                          data-testid="run-details-steps-panel"
                          key={selectedJobId}
                          jobId={selectedJobId}
                          viewportRef={stepWrapperRef}
                        />
                      )}
                    </TabPanel>
                    <TabPanel value={JOBS_SUB_TABS.EVIDENCE} sx={{ p: 0, height: tabListHeight }}>
                      <EvidencePanel
                        data-testid="run-details-evidence-panel"
                        jobId={selectedJobId}
                        artifacts={workflowArtifactsResponse?.artifacts}
                      />
                    </TabPanel>
                  </TabContext>
                </Box>
              </Panel>
            </PanelGroup>
          </Box>
        </TabPanel>
        <TabPanel value={TABS.TEST_RESULTS} sx={{ p: 0, height: tabListHeight }}>
          <Box sx={{ p: 2, height: '100%' }}>
            <TestResultsProvider
              testResultsLoading={testResultsLoading}
              isTestInsightsDataFound={testResultsData?.isTestInsightsDataFound}
            >
              <TestResults />
            </TestResultsProvider>
          </Box>
        </TabPanel>
      </TabContext>
    </Container>
  )
}

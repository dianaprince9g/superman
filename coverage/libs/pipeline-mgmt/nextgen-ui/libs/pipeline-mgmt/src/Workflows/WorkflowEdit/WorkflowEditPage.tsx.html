    componentId,
    automationId,

    /**
     * Parse the yaml here to ensure newest possible value, otherwise it can get stale or no data.

    const { parsed: parsedYaml } = parseYaml(prevYaml)

    const environmentAdded = jobs.some(currentJob => {

         * Check if environment was added to existing job.

        return !parsedYaml.jobs[currentJob.name].environment && currentJob.environment

      return !!currentJob.environment

    if (environmentAdded) {
      track('Workflow Environment Used', properties)

    setPrevYaml(yaml)
    notify(t('save-success'), 'success')

    const propertyValues = properties?.values as {
      branch_name: string
      file_name: string
    }
    const workflowIdFromCallback = (properties?.workflowId as string) || ''
    const newBranch = propertyValues.branch_name || branch
    const fileName = propertyValues.file_name || defaultFileName

    // If this is an initial creation (workflowId from params is undefined)
    // or if we save a file as a new workflow in different branch, we do want to reload the page
    // to fetch new params
    if (!workflowId || newBranch !== branch) {
      const path = generateWorkflowPath(
        workflowIdFromCallback,
        newBranch,
        domain,
        componentId,
        fileName,
      )
      navigate(path)
    }
  }
 
  const showErrorMessage = async (error: Response) => {
    const body = await error.json?.()

    const cases = {
      409: {
        'already exists': t('commit-drawer.name_exists'),
        'Update rejected, newer commits exist': t('commit-drawer.workflow-out-of-date'),
      },
    }

    if (cases[error.status as 409]?.[body?.message as 'already exists']) {
      return notify(cases[error.status as 409][body?.message as 'already exists'], 'error')
    }

    return notify(t('save-error'), 'error')
  }

    if (onSettledCB) {
      onSettledCB()
    }
    setShowDrawer(false)
  }
 
  const onSubmit: IAutomationCommitDrawerProps['onSubmit'] = async (
    values,
    onSettledCB?: () => void,
  ) => {
    const basePayload = {
      service_id: componentId,
      yaml_content: yaml,
      ...values,
    }

    const automationExists = !!automationId
    const callCreate = !automationExists || values.create_new_branch
    const payloadFinal = callCreate ? basePayload : { ...basePayload, commitSha }
    const automationFn = callCreate ? createAutomation : updateAutomation

    const onSuccess = (props: { automations: IAutomation[] }) => {
      const userId = user?.user?.['https://www.cloudbees.com/user/metadata']?.userId

      if (!userId) return

      // default values for existing workflow
      let workflowId = automationId
      let workflowName = values.file_name

      if (props.automations && props.automations.length > 0 && props.automations.at(0)) {
        // New workflow created
        workflowId = props.automations.at(0)!.id
        workflowName = props.automations.at(0)!.fileName
        track('Workflow Created', {
          workflowId,
          workflowName,
        })
      }

      automationSuccessCallback({
        workflowId,
        workflowName,
        values,
      })
    }
    try {
      automationFn(payloadFinal, {
        onSuccess: onSuccess,
        onError: showErrorMessage,
        onSettled: getOnSettledCallback(onSettledCB),
      })
    } catch (error) {
      if (ENV.NODE_ENV === 'development') console.error(error)

      notify(t('save-error'), 'error')
      setShowDrawer(false)
    }
  }
 
  const onCommit = async () => {
    if (parsingErrors.length > 0) {
      notify(t('invalid-yaml'), 'error')
    } else {
      setShowDrawer(true)
    }
  }
 
  const onCancel = () => {
    if (yaml !== prevYaml && ['false', false].includes(confirmModalSetting)) {
      setShowCancelConfirmModal(true)
    } else {
      onCancelConfirm()
    }
  }
 
  const onSplit = () => {
    setMode('SPLIT_VERTICAL')
  }
 
  const onScript = () => {
    setMode('SCRIPT_ONLY')
  }
 
  const onVisual = () => {
    setMode('VISUAL_ONLY')
  }
 
  // When there is not an automation ID, we are creating a new workflow and should not block navigation
  const isStale = yaml !== prevYaml && !!workflowId
 
  const onCancelConfirm = () => {
    setYaml(prevYaml)
    setShowCancelConfirmModal(false)
  }
 
  const usedTriggerTypes = Object.keys(trigger || {})
  const unusedTriggerTypes: ITriggerToDisplay[] = allTriggers.filter(
    item => !item.comingSoon && !usedTriggerTypes.includes(item.type),
  )
  const availableTriggers = unusedTriggerTypes.map(t => t.type)
  const comingSoonTriggers = allTriggers.filter(trigger => trigger.comingSoon)
 
  const componentBranches = useBranchesQuery(organizationId, componentId).data?.branches
  const [selectedTrigger, setSelectedTrigger] = useState<ISelectedTrigger>()
  const [selectedJob, setSelectedJob] = useState<IJobNode>()
 
  //parent of new job node
  const [parentNode, setParentNode] = useState<string | undefined>()
  //child of the new job node in case of adding job between 2 jobs
  const [targetNode, setTargetNode] = useState<string | undefined>()
  const [selectedStep, setSelectedStep] = useState<number>(-1)
  const [hover, setHover] = useState(false)

    Boolean(triggerId),
    componentId,
    triggerId,
  )
  const isYamlLoaded = Boolean(workflowsStream?.[0]?.job?.outputs?.[0]?.value) || !streamingLoading
 
  useEffect(() => {
    if (!streamingLoading && isYamlLoaded && workflowsStream) {
      const yamlWorkflow = workflowsStream?.[0]?.job?.outputs?.[0]?.value
      const yaml = yamlWorkflow ? JSON.parse(yamlWorkflow).workflow : ''

      setYaml(yaml)
      setPrevYaml(yaml)
    }
  }, [streamingLoading, isYamlLoaded, workflowsStream, setYaml, setPrevYaml])
 
  const onTriggerClick = useCallback((trigger?: ISelectedTrigger, triggerEditMode?: EditMode) => {
    doTriggerClick(
      setSelectedTrigger,
      setEditMode,
      setCreateTriggerDrawerOpen,
      trigger,
      triggerEditMode,
    )

  const onJobClick = useCallback((job: IJobNode, jobEditMode?: EditMode) => {
    doJobClick(
      setSelectedJob,
      setSelectedStep,
      setEditMode,
      setCreateJobDrawerOpen,
      job,
      jobEditMode,
    )
  }, [])

      deleteJob(job.label || '')

        type: 'JOB',
        action: Action.DELETE,

  )

    (label: string) => {
      deleteTrigger(label)

      track('Automation Graphical Composer Used', {
        type: 'TRIGGER',
        action: Action.DELETE,
      })

    [deleteTrigger, track],

    doCreateJob(
      setParentNode,

              onMouseEnter={() => setHover(true)}

                      <WorkflowFileName name={defaultFileName} />
                      <Link

                        fontSize={14}

                      >

                    onChipClick={() => {

                      onDragStart={event => onDragStart(event, 'NEW_MANUAL_APPROVAL')}

                      setDragType={setDragType}
                      backgroundColor={customComponents.Graph.manualApproval.borderColor as string}

            </Stack>
 
            <Stack
              height={1}
              sx={theme => ({
                'position': 'relative',
                'overflow': 'hidden',
                'borderRadius': '.25rem',
                'border': `1px solid ${
                  themeMode === 'dark' ? 'rgba(255, 255, 255, 0.25)' : '#0000001F'
                }`,
                'backgroundColor': theme.palette.background.paper,
                // Fix resize issue with monaco
                '.monaco-editor': {
                  position: 'absolute',
                },
              })}
            >
              <Box>
                <Box height={56} display={'flex'} alignItems={'center'} width={1} ml={3}>
                  <WorkflowFileName name={defaultFileName} />
                </Box>
              </Box>
              <CodeEditor
                language="yaml"
                value={yaml}
                onChange={value => setYaml(value)}
                onValidate={errors => setValidationErrors(errors)}
              />
            </Stack>
          </SplitView>
        </Box>
      </Container>
    </GraphActionsContext.Provider>
  )

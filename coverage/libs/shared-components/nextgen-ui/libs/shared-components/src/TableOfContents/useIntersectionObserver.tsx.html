import { useEffect, useRef } from 'react'
 
/**
 * Hook to observe the intersection of the headings
 *
 * @param headingElements - The heading elements to observe
 * @param setActiveId - Function to set the active id
 */
export const useIntersectionObserver = (
  headingElements: Element[],
  setActiveId: (id: string) => void,
) => {
  const headingElementsRef = useRef<Record<string, IntersectionObserverEntry>>({})
 
  useEffect(() => {
    const callback: IntersectionObserverCallback = headings => {
      headingElementsRef.current = headings.reduce((map, headingElement) => {
        map[headingElement.target.id] = headingElement
        return map
      }, headingElementsRef.current)

      const visibleHeadings: IntersectionObserverEntry[] = []
      Object.keys(headingElementsRef.current).forEach(key => {
        const headingElement = headingElementsRef.current[key]
        if (headingElement.isIntersecting) visibleHeadings.push(headingElement)
      })

      const getIndexFromId = (id: string) => headingElements.findIndex(heading => heading.id === id)

      if (visibleHeadings.length === 1) {
        setActiveId(visibleHeadings[0].target.id)
      } else if (visibleHeadings.length > 1) {
        const sortedVisibleHeadings = visibleHeadings.sort(
          (a, b) => getIndexFromId(a.target.id) - getIndexFromId(b.target.id),
        )

        setActiveId(sortedVisibleHeadings[0].target.id)
      }
    }
 
    const observer = new IntersectionObserver(callback, {
      rootMargin: '0% 0px -70% 0px',
    })
 
    headingElements.forEach(element => observer.observe(element))
 
    return () => observer.disconnect()
  }, [headingElements, setActiveId])
}

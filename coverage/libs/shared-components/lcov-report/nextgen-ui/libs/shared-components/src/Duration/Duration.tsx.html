import { useEffect, useState } from 'react'
import dayjs from 'dayjs'
import duration from 'dayjs/plugin/duration'
 
dayjs.extend(duration)
 
// Times are in unix time
// Zero time means there is no data
export interface IDurationProps {
  startTime: number // in second
  endTime: number // in second
}
 
export const calculateDurationString = ({ startTime, endTime }: IDurationProps): string => {
  const startTimeMilliseconds = startTime * 1000
  const endTimeMilliseconds = endTime === 0 ? Date.now() : endTime * 1000
 
  // Formats in 8d 8h 8m 8s format.  Removes zero values.
  return dayjs
    .duration(endTimeMilliseconds - startTimeMilliseconds)
    .format(' D[d] H[h] m[m] s[s]')
    .replace(/\s0[a-z]/g, '') // replaces " 0d" or " 0h" or " 0m" or " 0s" with empty string
    .trim()
}
 
/**
 * Displays unix start and end times as a duration in "<hours>h <minutes>m <seconds>s" format.
 * Numbers never begin with a zero.  If there is no end time (if end time is 0), the duration will update every second.
 *
 * @param props
 * @returns
 */
export const DurationComponent = (props: IDurationProps) => {
  const [duration, setDuration] = useState<string>(calculateDurationString(props))
 
  useEffect(() => {
    // Only run the interval if there is no end time set
    if (props.endTime === 0) {
      const interval = setInterval(() => {
        setDuration(calculateDurationString(props))
      }, 1000)

      return () => clearInterval(interval)
    }
 
    return
  }, [duration, props])
 
  return <span>{duration}</span>
}
 
export const Duration = DurationComponent

const t = (key: string) =>
  i18next.t(`connectors.create-connector-form.${key}`, { ns: 'platform-settings' })
 
const TYPE_CATEGORY_MAP: Record<string, { type: string; category: string }> = {
  'github-app-endpoint-type': {
    type: t('step-descriptions.github-app'),
    category: t('select-category.categories.code-repository'),
  },
  'github-token-endpoint-type': {
    type: t('step-descriptions.github-access-token'),

  'jira-token-endpoint-type': {
    type: t('step-descriptions.jira-access-token'),

  },
}
 
export const useConnectorsQuery = (
  resourceId?: string,
  enabled = true,
  interval: false | number | undefined = false,
) => {
  const queryResponse = useQueryGRApiPlatform<IConnectorsResponse>({
    queryKey: ['endpoints', resourceId],
    path: `/v1/resources/${resourceId}/endpoints?&filter.contributionTypes=cb.platform.endpoint-type&parents=true`,
    method: 'GET',
    enabled: enabled && !!resourceId,
    refetchInterval: interval,
  })
  if (queryResponse.data) {
    queryResponse.data.endpoints = queryResponse.data.endpoints?.map(connector => {
      return {
        ...connector,
        type: TYPE_CATEGORY_MAP[connector.contributionId]?.type ?? connector.contributionId,
        category: TYPE_CATEGORY_MAP[connector.contributionId]?.category ?? '—',
      }
    })
  }
  return queryResponse
}
 
const getCategories = (manifest: IManifest) => {
  const categories = manifest.contributions.map(contribution => {
    return contribution.spec?.categories || ''
  })
  return [...new Set(categories.flat())]
}
 
export const useExtensionsCatalog = () => {
  const orgId = useOrganizationId()
  const { data } = useQueryGRApiPlatform<IProvidersResponse>({
    queryKey: ['extensions-catalog'],
    path: `/v1/organizations/${orgId}/extensions`,
    method: 'GET',
  })
  return data
}

export const useConnectorsProviders = (): IProvider[] => {
  const data = useExtensionsCatalog()
  return data && data.manifests
    ? data.manifests.map(manifest => {
        return {
          id: manifest.id,
          name: manifest.name,
          iconUrl: manifest.contributions[0]?.spec?.iconUrl ?? 'internal:/icons/unknown',
          categories: getCategories(manifest),
          providerTypes: manifest.contributions.map(contribution => {
            return {
              id: contribution.id,
              description: contribution.description,
              title: contribution.spec?.displayName || '',
              recommended: contribution.spec?.recommended ?? false,
              apiDescriptors: contribution.spec?.apiDescriptors,
            }
          }),
        }
      })
    : []
}
 
function findContributionById(manifests: IManifest[], contributionId: string) {
  for (const manifest of manifests) {
    for (const contribution of manifest.contributions) {
      if (contribution.id === contributionId) {
        return contribution
      }
    }
  }
  return null
}
 
export const useConfigurationDescriptors = (contributionId: string): IConfigurationDescriptor[] => {
  const data = useExtensionsCatalog()

  const selectedContribution =
    data?.manifests && findContributionById(data.manifests, contributionId)

  return selectedContribution?.spec?.configurationDescriptors ?? []
}
 
export const useConnectionTestDescriptor = (
  contributionId: string,
): IConnectionTest | undefined => {
  const data = useExtensionsCatalog()

  const selectedContribution =
    data?.manifests && findContributionById(data.manifests, contributionId)

  return selectedContribution?.spec?.connectionTest?.[0]
}

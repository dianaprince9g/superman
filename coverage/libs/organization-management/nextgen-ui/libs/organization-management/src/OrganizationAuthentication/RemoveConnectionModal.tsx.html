import { useQueryClient } from '@tanstack/react-query'
 
import { useApiMutation } from '@cloudbees/api-platform/queries/guiderails'
import { useAuthenticatedUser } from '@cloudbees/shared/user'
import { ConfirmationModal } from '@cloudbees/shared-components/ConfirmationModal'
import { useNotification } from '@cloudbees/shared-hooks/useNotification'
import { type IPlatformConnection } from '@cloudbees/shared-types/types/platform_sso'
 
interface IRemoveConnectionModal {
  open: boolean
  connection: IPlatformConnection
  onClose: () => void
}
 
export const RemoveConnectionModal = ({ connection, onClose, open }: IRemoveConnectionModal) => {
  const { notify } = useNotification()
  const { organization } = useAuthenticatedUser()
 
  const { DELETE } = useApiMutation()
 
  const queryClient = useQueryClient()
 
  const removeConnection = async () => {
    try {
      await DELETE(`/v1/organizations/${organization?.id}/saml/connections/${connection?.id}`)
      notify(`Connection ${connection.connectionName} removed successfully`, 'success')
      queryClient.invalidateQueries({ queryKey: ['authentication', 'connections'] })
    } catch {
      notify(`Failed to remove ${connection.connectionName}`, 'error')
    } finally {
      onClose()
    }
  }
 
  if (!connection || !organization) {
    return null
  }
 
  return (
    <ConfirmationModal.Modal onClose={onClose} aria-labelledby="remove-domain-modal" open={open}>
      <ConfirmationModal.Title id="customized-dialog-title">
        Remove connection
      </ConfirmationModal.Title>
      <ConfirmationModal.Content>
        <ConfirmationModal.ContentText>
          Removing the {connection.connectionName} connection also removes all associated domains.
        </ConfirmationModal.ContentText>
      </ConfirmationModal.Content>
      <ConfirmationModal.Actions>
        <ConfirmationModal.ActionItem data-testid="cancel-button" onClick={onClose}>
          Cancel
        </ConfirmationModal.ActionItem>
        <ConfirmationModal.ActionItem
          data-testid="remove-button"
          autoFocus
          color="error"
          onClick={removeConnection}
        >
          Remove
        </ConfirmationModal.ActionItem>
      </ConfirmationModal.Actions>
    </ConfirmationModal.Modal>
  )
}

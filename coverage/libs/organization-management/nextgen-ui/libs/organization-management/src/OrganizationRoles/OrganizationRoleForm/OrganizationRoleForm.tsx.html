import { useState } from 'react'
import { useTranslation } from 'react-i18next'
import { generatePath } from 'react-router-dom'
import Box from '@mui/material/Box'
import Button from '@mui/material/Button'
import { object, string } from 'yup'
 
import { ROLES_PATH } from '@cloudbees/route-config-nextgen'
import { useAuthenticatedUser } from '@cloudbees/shared/user'
import { ConfirmationModal } from '@cloudbees/shared-components/ConfirmationModal'
import { useNotification } from '@cloudbees/shared-hooks/useNotification'
import { Form } from '@cloudbees/shared-patterns/Form'
import { PageHeader } from '@cloudbees/shared-patterns/PageHeader'
import { type IModifiedRolePermission } from '@cloudbees/shared-types'
import { useNavigate, useParams } from '@cloudbees/shared-utils/navigation'
import { usePreferences } from '@cloudbees/user-management/Preferences/context/PreferencesContext'
 
import { type Callbacks, type CreateRoleRequest } from '../OrganizationCreateRolePage'
import { OrganizationRolePage } from '../OrganizationRolePage'
 
import { OrganizationCreateRoleEditableField } from './OrganizationCreateRoleEditableField'
 
interface Actions {
  [action: string]: boolean
}
 
interface Roles {
  [roleType: string]: Actions
}
 
interface RolePermission {
  action: string
  type: string
}
 
interface CreateRoleFormValues {
  roleName: string
  description: string
  roles: Roles
}
 
interface IOrganizationRoleFormProps {
  onSubmit: (variables: CreateRoleRequest, options: Callbacks) => Promise<void>
  role?: IModifiedRolePermission
}
 
const convertRolesToPermissions = (roles: Roles): RolePermission[] =>
  Object.entries(roles).reduce((acc: RolePermission[], [type, actions]) => {
    Object.entries(actions).forEach(([action, isAllowed]) => {
      if (isAllowed) {
        acc.push({ action, type })
      }
    })
    return acc
  }, [])
 
export const OrganizationRoleForm = ({ onSubmit, role }: IOrganizationRoleFormProps) => {
  const { t } = useTranslation('organization-management', {
    keyPrefix: 'organization-roles',
  })
  const { domain } = useAuthenticatedUser()
  const { notify } = useNotification()
  const { formatDate } = usePreferences()
 
  const [isCancelModalOpen, setIsCancelModalOpen] = useState<boolean>(false)
  const navigate = useNavigate()
  const { orgId = null } = useParams()
 
  const redirectToRolesPage = () => {
    const path = generatePath(ROLES_PATH, { orgId })
    navigate(path)
  }
 
  const validationSchema = object().shape({
    roleName: string()
      .required(t('validation.role-required'))
      .min(3, t('validation.role-min'))
      .max(50, t('validation.role-max')),
  })
 
  const handleSumbit = async (formValues: CreateRoleFormValues) => {
    const organizationId = domain?.id || ''
    const payload = {
      role: {
        organizationId,
        name: formValues.roleName,
        permissions: convertRolesToPermissions(formValues.roles),
        description: formValues.description,
      },
      organizationId,
    }
    const onError = (err: Response) => {
      if (err.status === 409) {
        notify(t('role.duplicate-error'), 'error')
      } else {
        notify(t('role.error'), 'error')
      }
    }
    const onSuccess = () => {
      redirectToRolesPage()
      notify(t(role ? 'role.modified' : 'role.success'), 'success')
    }
    await onSubmit(payload, { onSuccess, onError })
  }
 
  return (
    <Form.Context
      initialValues={{
        roleName: role?.name || 'Custom Role',
        description: role?.description || '',
        roles: role?.permissions || {},
      }}
      validationSchema={validationSchema}
      onSubmit={handleSumbit}
    >
      {({ values, setFieldValue }) => (
        <div>
          <PageHeader.Wrapper>
            <ConfirmationModal.Modal open={isCancelModalOpen} fullWidth={false} maxWidth="sm">
              <ConfirmationModal.Title id="customized-dialog-title">
                {t(`confirmation_modal.title`)}
              </ConfirmationModal.Title>
              <ConfirmationModal.Content>
                <ConfirmationModal.ContentText>
                  {t(`confirmation_modal.content-text`)}
                </ConfirmationModal.ContentText>
              </ConfirmationModal.Content>
              <ConfirmationModal.Actions>
                <ConfirmationModal.ActionItem
                  data-testid="custom-role-cancel"
                  autoFocus
                  onClick={() => setIsCancelModalOpen(false)}
                >
                  {t(`confirmation_modal.go-back`)}
                </ConfirmationModal.ActionItem>
                <ConfirmationModal.ActionItem
                  data-testid="delete-confirmation"
                  variant="contained"
                  onClick={redirectToRolesPage}
                >
                  {t(`confirmation_modal.cancel`)}
                </ConfirmationModal.ActionItem>
              </ConfirmationModal.Actions>
            </ConfirmationModal.Modal>
            <PageHeader.Actions>
              <Button variant="outlined" onClick={() => setIsCancelModalOpen(true)}>
                {t(`editable-fields.cancel`)}
              </Button>
              <Form.Submit> {t(`editable-fields.save`)}</Form.Submit>
            </PageHeader.Actions>
            <OrganizationCreateRoleEditableField
              editableFieldName={'roleName'}
              label={t('editable-fields.role-label')}
              value={role?.name || values.roleName}
            />
            <OrganizationCreateRoleEditableField
              editableFieldName={'description'}
              label={t('editable-fields.description-label')}
              value={role?.description || values.description}
            />
            <Box display="flex" alignItems="center" gap={3}>
              {!role && (
                <PageHeader.Description>{`${t('role.created-date')} ${formatDate(
                  new Date(),
                  'date',
                )}`}</PageHeader.Description>
              )}
              <PageHeader.Description>{`${t('role.last-modified')} ${
                role ? formatDate(role?.audit?.when ?? '') : formatDate(new Date(), 'date')
              }`}</PageHeader.Description>
            </Box>
          </PageHeader.Wrapper>
          <OrganizationRolePage parentValue={values} setFieldValue={setFieldValue} />
        </div>
      )}
    </Form.Context>
  )
}

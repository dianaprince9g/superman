import { useState } from 'react'
 
import { useGetDomains } from '@cloudbees/api-platform/queries/authentication/useGetDomains'
import { useInstrumentation } from '@cloudbees/api-platform/queries/useInstrumentation'
import { useAuthenticatedUser } from '@cloudbees/shared/user'
import { CircularProgressCenterLoader } from '@cloudbees/shared-components/CircularProgressCenterLoader'
import { ZeroState } from '@cloudbees/shared-components/ZeroState'
import { useAppContext } from '@cloudbees/shared-providers/components/AppContextProvider'
import { type IPlatformConnection } from '@cloudbees/shared-types/types/platform_sso'
 
import { SamlConfigurationDrawer } from './SamlDrawer/SamlConfigurationDrawer'
import { type IDomainsTableData, type ISSOPlatformState } from './AuthenticationDomainsTable'
import { AuthenticationHeader } from './AuthenticationHeader'
import { AuthenticationTabs } from './AuthenticationTabs'
import { RemoveConnectionModal } from './RemoveConnectionModal'
import { RemoveDomainModal } from './RemoveDomainModal'
 
export const AuthenticationPage = () => {
  const { track } = useInstrumentation()
  const user = useAuthenticatedUser()
 
  const { isLoading: domainsLoading, data: domainsData, refetch: refetchDomains } = useGetDomains()
 
  const [showDrawer, setShowDrawer] = useState(false)
  const [showRemoveDomainModal, setShowRemoveDomainModal] = useState(false)
  const [showRemoveConnectionModal, setShowRemoveConnectionModal] = useState(false)
 
  const [startingStep, setStartingStep] = useState<number>(0)
 
  const [currentDomain, setCurrentDomain] = useState<IDomainsTableData | undefined>(undefined)
  const [currentConnection, setCurrentConnection] = useState<IPlatformConnection | undefined>(
    undefined,
  )
 
  const handleOpenAddDomainFormAtStep = (ssoState: ISSOPlatformState) => {
    setStartingStep(ssoState.step)
    setCurrentDomain(ssoState.domain)
    setCurrentConnection(ssoState.connection)
    setShowDrawer(true)
  }
 
  const handleRemoveDomain = (domain: IDomainsTableData) => {
    setShowRemoveDomainModal(true)
    setCurrentDomain(domain)
  }
 
  const handleRemoveConnection = (connection: IPlatformConnection) => {
    setShowRemoveConnectionModal(true)
    setCurrentConnection(connection)
  }
 
  const openSamlConfigurationDrawer = () => {
    setStartingStep(0)
    setCurrentDomain(undefined)
    setCurrentConnection(undefined)
    setShowDrawer(true)
    track('SAML Setup Started', {
      tenantId: user.domain?.id,
    })
  }
  const closeDrawer = (step?: number) => {
    track('SAML Setup Cancelled', {
      tenantId: user.domain?.id,
      step,
    })
    setShowDrawer(false)
    refetchDomains()
  }
 
  const app = useAppContext()
 
  if (domainsLoading || !domainsData) {
    return <CircularProgressCenterLoader data-testid="auth" />
  }
 
  return (
    <>
      <AuthenticationHeader
        hideCreateSamlButton={app === 'admin-tools' || domainsData.emailDomains.length === 0}
        openCreateSamlDrawer={() => openSamlConfigurationDrawer()}
      />
      {domainsData.emailDomains.length > 0 ? (
        <AuthenticationTabs
          openAddDomainFormAtStep={handleOpenAddDomainFormAtStep}
          handleRemoveDomain={handleRemoveDomain}
          handleRemoveConnection={handleRemoveConnection}
        />
      ) : (
        <ZeroState
          onClick={() => openSamlConfigurationDrawer()}
          title="Create a SAML single sign-on"
          description="Verify domain ownership, establish a mutual connection with your identity provider, and then activate single sign-on."
          ctaText="Create SAML"
        />
      )}
      <SamlConfigurationDrawer
        initialStep={startingStep}
        showDrawer={showDrawer}
        refetchDomains={refetchDomains}
        closeDrawer={step => closeDrawer(step)}
        domain={currentDomain}
        connection={currentConnection}
      />
      <RemoveDomainModal
        open={showRemoveDomainModal}
        domain={currentDomain}
        onClose={() => setShowRemoveDomainModal(false)}
      />
      {currentConnection && (
        <RemoveConnectionModal
          open={showRemoveConnectionModal}
          connection={currentConnection}
          onClose={() => setShowRemoveConnectionModal(false)}
        />
      )}
    </>
  )
}

import { entity } from 'simpler-state'
 
import { type IFetchOptions } from '@cloudbees/shared-types'
 
/* The write action modal can have 3 main views (in the order defined below) */
 
export enum WriteActionView {
  ZENDESK_TICKET_PROMPT = 'ZENDESK_TICKET_PROMPT',
  CUSTOM_MODAL = 'CUSTOM_MODAL',
  FINAL_CONFIRMATION = 'FINAL_CONFIRMATION',
}
 
/*
  Through the properties in state we can allow each modal to
  customize the messages, sub-components, and actions of the global WriteActionModal
*/
 
export interface IWriteActionModalState {
  isOpen: boolean
  onClose: () => void
  title?: string
 
  // Control the view that the modal shows
  view: WriteActionView
 
  // Custom modal that we might show to user (to collect more input about an action)
  customModal?: JSX.Element
 
  // The "write" action that the modal will ultimately execute.
  writeAction: (options: IFetchOptions) => any
 
  // Message to display in the final confirmation view
  confirmationMessage?: JSX.Element
 
  // Messages to show in success/error toast, after the modal attempts to perform the writeAction.
  successMessage?: string
  failureMessage?: string
 
  // By default, when the WriteActionModal receives a WriteAction response, it'll
  // display an error message if the response code is not OK. You can override that behavior by
  // passing onFailureCallback function. This is useful if you want to raise specific toasts based on a given response code.
  onFailureCallback?: (response: Response) => void
 
  // By default, when the WriteActionModal receives a WriteAction response, it'll
  // display a succcess message if the response code is OK. You can override that behavior by
  // passing onSuccessCallback.
  // It may be useful to do something after the network call the modal makes.
  // For example, oftentimes we need to refetch data after creating or updating data via a modal.
  onSuccessCallback?: () => void
}
 
/*
  Using simpler-state's entity was the easiest choice for this global state.
 
  The alternatives were:
    * Redux - Several files for state + dispatch + reducers
    * A funky combination of ContextProvider + React Reducer - Complicated...
*/
 
const defaultState: IWriteActionModalState = {
  isOpen: false,
  onClose: () => undefined,
  title: '',
  view: WriteActionView.ZENDESK_TICKET_PROMPT,
  customModal: undefined,
  writeAction: (customHeaders: any) => undefined,
  confirmationMessage: undefined,
  successMessage: '',
  failureMessage: '',
  onFailureCallback: undefined,
  onSuccessCallback: undefined,
}
 
export const writeActionModalState = entity(defaultState)
 
/* These methods provide an API for WriteActionModal singleton */
 
export const WriteActionModalAPI = {
  /**
   * Binds the global modal's state to the caller component.
   *
   * This is equivalent as calling `Entity().use()` from `simpler-state` module.
   *
   * @return {IWriteActionModalState} A read-only instance of the global state.
   */
  use(): IWriteActionModalState {
    return writeActionModalState.use()
  },
 
  /**
   * Updates WriteActionModal's global state. Allows updating partial properties as well.
   *
   * @param {IWriteActionModalState} newModalState - The new state object to update with (or a partial of it).
   */
  setupModal(newModalState: Partial<IWriteActionModalState>) {
    writeActionModalState.set((currentState: IWriteActionModalState) => {
      return {
        ...currentState,
        ...newModalState,
      }
    })
  },
 
  openModal() {
    WriteActionModalAPI.setupModal({ isOpen: true })
  },
 
  closeModal() {
    WriteActionModalAPI.setupModal(defaultState)
  },
 
  advanceToCustomModalView() {
    WriteActionModalAPI.setupModal({ view: WriteActionView.CUSTOM_MODAL })
  },
 
  advanceViewToFinalConfirmation() {
    WriteActionModalAPI.setupModal({ view: WriteActionView.FINAL_CONFIRMATION })
  },
}

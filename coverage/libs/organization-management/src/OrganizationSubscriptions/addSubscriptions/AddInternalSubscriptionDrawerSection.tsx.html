import { useCallback, useMemo } from 'react'
import { useTranslation } from 'react-i18next'
import flatten from 'lodash/flatten'
import { object, string } from 'yup'
 
import {
  type IEntitlement,
  type ISubscription,
  type ISubscriptionData,
} from '@cloudbees/shared-auth/types'
import { useDrawerStepperContext } from '@cloudbees/shared-patterns/DrawerStepper'
import { Form } from '@cloudbees/shared-patterns/Form'
import { type ISku, type ISkuData } from '@cloudbees/shared-types'
 
import { SubscriptionPlanSelect } from './SubscriptionPlanSelect'
 
interface IAddInternalSubscriptionDrawerSectionProps {
  skus: ISku
  subscriptionData: ISubscription
  organizationId: string
}
 
interface ProductFamilyInfo {
  productFamilyName: string
  description: string
}
 
export interface AddInternalSubscriptionFormData {
  product: string
  plan: string
}
 
export const AddInternalSubscriptionDrawerSection = (
  props: IAddInternalSubscriptionDrawerSectionProps,
) => {
  const { t } = useTranslation('organization-management', {
    keyPrefix: 'organization-subscriptions.modal',
  })
 
  /* Convert SKUs object to array */
 
  const arrayOfSkuData = useMemo(() => {
    return Object.entries(props.skus ?? {})
  }, [props.skus])
 
  const listOfAllInternalSkus = useMemo(() => {
    return arrayOfSkuData.map((x: [string, ISkuData]) => x[0])
  }, [arrayOfSkuData])
 
  /* Get the list of product families that we'll show in the 1st dropdown */
 
  const listOfAllProductFamilies = useMemo((): Array<string> => {
    const allProducts = arrayOfSkuData.map((x: [string, ISkuData]) => {
      return {
        productFamilyName: x[1].productFamily,
        description: x[1].entitlements.display_name,
      } as ProductFamilyInfo
    })
 
    const setOfUniqueProducFamilies = new Set<string>()
    allProducts.forEach((x: ProductFamilyInfo) =>
      setOfUniqueProducFamilies.add(x.productFamilyName),
    )
 
    // Add the header of the dropdown (The default option before anything is selected)
    const listOfUniqueProductFamilies = Array.from(setOfUniqueProducFamilies)
 
    return listOfUniqueProductFamilies
  }, [arrayOfSkuData])
 
  /* Method to get the existent entitlements, used to later update all entitlements */
 
  const getCurrentlyAppliedInternalEntitlements = useCallback(() => {
    const entitlements = props.subscriptionData.map((subscription: ISubscriptionData) => {
      return subscription.entitlements.map((z: IEntitlement) => {
        return z
      })
    })
 
    // props.subscriptionData contains all subscriptions (internal and external), when doing a POST
    // to subscription service, we only want to POST the internal entitlements.
    return flatten(entitlements).filter((entitlement: IEntitlement) =>
      arrayOfSkuData.filter((x: [string, ISkuData]) => x[0] === entitlement.sku),
    )
  }, [arrayOfSkuData, props.subscriptionData])
 
  const { setData, nextStep, nextButton } = useDrawerStepperContext()
 
  const addSubscription = async (values: AddInternalSubscriptionFormData) => {
    const accountId = props.subscriptionData.find(
      (x: ISubscriptionData) => x.provider === 'CloudBees Other',
    )?.id
    const relevantSku = arrayOfSkuData.find(
      (x: [string, ISkuData]) =>
        x[1].productFamily === values.product && x[1].entitlements.display_name === values.plan,
    )
    const skuKey = relevantSku?.[0]
 
    const fullEntitlements: Array<IEntitlement | Partial<IEntitlement>> = []
 
    getCurrentlyAppliedInternalEntitlements().forEach((x: IEntitlement) => {
      if (listOfAllInternalSkus.includes(x.sku)) {
        // The SKU we're posting to the API must exist in the list of internal SKUs. Otherwise,
        // it's an external entitlement which the subscription service cannot handle.
        fullEntitlements.push(x)
      }
    })
 
    // note: many of these fields might be set later, leaving for now
    fullEntitlements.push({
      sku: skuKey,
      metadata: {},
      unit: 1,
      startDate: new Date('2021-08-12T17:16:44.468Z'),
      endDate: new Date('2021-08-12T17:16:44.468Z'),
      accountId,
    })
 
    setData({
      path: accountId ? `/v1/accounts/${accountId}` : `/v1/accounts/`,
      method: accountId ? 'PUT' : 'POST', // POST if account doesn't exist, otherwise PUT
      body: JSON.stringify({
        externalAccountId: props.organizationId,
        organizationId: props.organizationId,
        provider: 'CloudBees Other',
        providerUrl: 'unknown',
        entitlements: fullEntitlements,
      }),
    })
    nextStep(1)
  }
 
  const validationSchema = object({
    product: string().required(),
    plan: string().required(),
  })
 
  return (
    <Form.Context<AddInternalSubscriptionFormData>
      initialValues={{ product: '', plan: '' }}
      validationSchema={validationSchema}
      onSubmit={async values => {
        return await addSubscription(values)
      }}
    >
      <Form.Form>
        <Form.Select
          dropdownItems={listOfAllProductFamilies}
          id="product"
          label={t('product')}
          placeholder={t('select-product') ?? ''}
          sx={{ mb: 2 }}
        />
        <SubscriptionPlanSelect skuData={arrayOfSkuData} />
        {nextButton(
          <Form.Submit data-testid="add-internal-subscription-submit-button">
            {t('add-internal-subscription')}
          </Form.Submit>,
        )}
      </Form.Form>
    </Form.Context>
  )
}

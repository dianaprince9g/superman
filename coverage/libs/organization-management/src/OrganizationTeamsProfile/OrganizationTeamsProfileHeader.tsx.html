import { useTranslation } from 'react-i18next'
import Button from '@mui/material/Button'
import Chip from '@mui/material/Chip'
 
import {
  useOrganizationContext,
  useOrganizationId,
} from '@cloudbees/shared-auth/RequireOrganization'
import { PageHeader } from '@cloudbees/shared-patterns/PageHeader'
import { useAppContext } from '@cloudbees/shared-providers'
import { useGetFeatureFlags } from '@cloudbees/shared-rox/useGetFeatureFlags'
import { type IPlatformTeam } from '@cloudbees/shared-types'
 
interface IOrganizationTeamsProfileHeaderProps {
  team: IPlatformTeam
  onTeamPublish: () => void
  onTeamEdit: () => void
}
 
export const OrganizationTeamsProfileHeader = ({
  team,
  onTeamPublish,
  onTeamEdit,
}: IOrganizationTeamsProfileHeaderProps) => {
  const { t: gt } = useTranslation('global', {
    keyPrefix: 'breadcrumbs',
  })
  const { t } = useTranslation('organization-management', {
    keyPrefix: 'organization-teams-profile.header',
  })
 
  const { platform } = useGetFeatureFlags()
  const showPublishButtons = platform.showPublishButtons.isEnabled()
 
  const isAdminTools = useAppContext() === 'admin-tools'
 
  const organization = useOrganizationContext()
  const orgId = useOrganizationId()
 
  return (
    <PageHeader.Wrapper>
      {isAdminTools && (
        <PageHeader.Breadcrumbs>
          <PageHeader.BreadcrumbItem href={'/'}>{gt('home')}</PageHeader.BreadcrumbItem>
 
          <PageHeader.BreadcrumbItem href="/organizations">
            {gt('organizations')}
          </PageHeader.BreadcrumbItem>
          <PageHeader.BreadcrumbItem>{organization?.name}</PageHeader.BreadcrumbItem>
          <PageHeader.BreadcrumbItem href={`/organizations/${orgId}/organization-teams`}>
            {gt('organization-teams')}
          </PageHeader.BreadcrumbItem>
          <PageHeader.BreadcrumbItem current>{gt('team')}</PageHeader.BreadcrumbItem>
        </PageHeader.Breadcrumbs>
      )}
 
      <PageHeader.Title>
        {team.name}
        {team.metadata && !!team.metadata.zdSupportEnabled && (
          <Chip sx={{ ml: 2 }} label={t('support-enabled')} color="success" />
        )}
      </PageHeader.Title>
      {team.metadata?.description && (
        <PageHeader.Description>{team.metadata.description}</PageHeader.Description>
      )}
      <PageHeader.Actions>
        {isAdminTools && showPublishButtons && (
          <Button variant="outlined" onClick={onTeamPublish} data-testid="publish-team-btn">
            {t('publish-team-button')}
          </Button>
        )}
        <Button variant="contained" onClick={onTeamEdit} data-testid="edit-team-btn">
          {t('edit-team')}
        </Button>
      </PageHeader.Actions>
    </PageHeader.Wrapper>
  )
}

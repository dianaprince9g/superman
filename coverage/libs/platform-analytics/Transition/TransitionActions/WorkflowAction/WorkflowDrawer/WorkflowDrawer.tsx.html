import { type Dispatch, type SetStateAction, useMemo, useState } from 'react'
import { useTranslation } from 'react-i18next'
import { generatePath, useSearchParams } from 'react-router-dom'
import Autocomplete from '@mui/material/Autocomplete'
import Divider from '@mui/material/Divider'
import FormControl from '@mui/material/FormControl'
import TextField from '@mui/material/TextField'
import Typography from '@mui/material/Typography'
import DOMPurify from 'dompurify'
 
import { useComponentsAutomationsQuery } from '@cloudbees/api-platform/queries/useComponentsAutomationsQuery'
import { COMPOSER_PATH } from '@cloudbees/route-config-nextgen'
import { useOrganizationId } from '@cloudbees/shared-auth/RequireOrganization'
import { ConfirmationModal } from '@cloudbees/shared-components/ConfirmationModal'
import { useColorMode } from '@cloudbees/shared-theme/UIProvider'
import { useNavigate, useParams } from '@cloudbees/shared-utils/navigation'
 
interface IWorkflowDrawer {
  setOpenWorkflowDrawer: Dispatch<SetStateAction<boolean>>
  showHeaderInfo?: string
}
 
interface IComponentOptions {
  id: string
  name: string
}
 
export const WorkflowDrawer = ({ setOpenWorkflowDrawer, showHeaderInfo }: IWorkflowDrawer) => {
  const { t } = useTranslation('platform-analytics', { keyPrefix: 'transition' })
  const { mode } = useColorMode()
  const { orgId = '', componentId = '' } = useParams<{ orgId: string; componentId: string }>()
  const [selectedComponent, setSelectedComponent] = useState<IComponentOptions | null>({
    id: componentId || '',
    name: '',
  })
 
  const organizationId = useOrganizationId()
  const [searchParams] = useSearchParams()
  const navigate = useNavigate()
 
  const { data: componentsResponse } = useComponentsAutomationsQuery({
    organizationId: organizationId,
  })
 
  const components = useMemo(
    () =>
      componentsResponse?.services?.map((component: IComponentOptions) => ({
        id: component?.id,
        name: component?.name,
      })),
    [componentsResponse],
  )
 
  const handleGotoWorkflowClick = () => {
    setOpenWorkflowDrawer(false)
    searchParams.set(
      'componentId',
      (selectedComponent?.id || components?.find(component => component)?.id) ?? '',
    )
    navigate({
      pathname: generatePath(COMPOSER_PATH, {
        orgId,
        branch: null,
      }),
      search: searchParams.toString(),
    })
  }
 
  return (
    <ConfirmationModal.Modal open maxWidth="sm">
      <ConfirmationModal.Content>
        {showHeaderInfo && (
          <Typography
            sx={{
              background: mode === 'dark' ? '#2E2E2E' : '#F8F8F8',
              borderRadius: 2,
              padding: 2,
              mt: 2,
            }}
            dangerouslySetInnerHTML={{ __html: DOMPurify.sanitize(showHeaderInfo) }}
          />
        )}
        {!componentId && components?.length > 1 && (
          <>
            <Typography sx={{ paddingTop: 3 }}>{t('select-component')}</Typography>
            <FormControl fullWidth sx={{ pt: 2, pb: 2 }}>
              <Autocomplete
                value={selectedComponent}
                id="component"
                options={components}
                getOptionLabel={option => option?.name}
                renderInput={params => <TextField {...params} label="Component" />}
                onChange={(e, value) => setSelectedComponent(value)}
              />
            </FormControl>
          </>
        )}
      </ConfirmationModal.Content>
      <Divider />
      <ConfirmationModal.Actions sx={{ padding: 2 }}>
        <ConfirmationModal.ActionItem onClick={() => setOpenWorkflowDrawer(false)}>
          {t('cancel')}
        </ConfirmationModal.ActionItem>
        <ConfirmationModal.ActionItem
          disabled={components?.length > 1 && (!selectedComponent || selectedComponent?.id === '')}
          variant="contained"
          onClick={handleGotoWorkflowClick}
        >
          {t('workflow.actionLabel')}
        </ConfirmationModal.ActionItem>
      </ConfirmationModal.Actions>
    </ConfirmationModal.Modal>
  )
}

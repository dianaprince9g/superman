const t = (key: string) => i18next.t(`filters.${key}`, { ns: 'shared-patterns' })
const AppliedFilters = () => {
  const { appliedFilters } = useCIInsightsContext()
  return (
    <Box sx={{ pl: 2 }}>
      <InputLabel>{t('appliedFilters')}</InputLabel>
      {Object.keys(appliedFilters).length > 0 && (
        <Box sx={{ display: 'flex', flexWrap: 'wrap' }}>
          {Object.keys(appliedFilters).map((appliedFilter, index) => {
            const optionText = appliedFilters[appliedFilter]?.optionText?.toLocaleString()
            return (
              <InputLabel key={`appliedFilter-${index}`} sx={{ pr: 2, maxWidth: '500px' }}>
                {`${appliedFilters[appliedFilter].title}: `}
                <Tooltip title={optionText}>
                  <b>{optionText}</b>
                </Tooltip>
              </InputLabel>
            )
          })}
        </Box>
      )}
    </Box>
  )
}
const FilterContent = ({ filters }: { filters: ICIInsightsFilter[] }) => {
  const renderContentFilter = useCallback(
    (filter: ICIInsightsFilter, index: number) => (
      <Grid item key={`content-${index}`} pb={2} pt={1}>
        {Filters(filter)}
      </Grid>
    ),
    [filters],
  )
  return <>{filters?.length > 0 && filters.map(renderContentFilter)}</>
}
const Filters = (filter: ICIInsightsFilter) => {
  const { type } = filter
  const filterTypes = {
    [SELECT]: () => DashboardFilterSelect(filter),
    [DURATION_SELECT]: () => DashboardFilterDurationSelect(filter),
    default: () => null,
  }
  const renderFilter = filterTypes[type] || filterTypes.default
  return renderFilter()
}
 
export const DashboardFilterSelect = ({
  label,
  name,
  type,
  options,
  initialValue,
}: ICIInsightsFilter) => {
  const {
    updateFilterValue,
    getOptionText,
    getFilterValue,
    getToolType,
    getStatus,
    getErrorConfiguredTime,
  } = useCIInsightsContext()
  const handleChange = (event: ISelectChangeEvent<string>) => {
    const { value } = event.target
    const optionText = getOptionText(type, options, value)
    const toolType = getToolType(options, value)
    const status = getStatus(options, value)
    const errorConfiguredTime = getErrorConfiguredTime(options, value)
    updateFilterValue(name, value, optionText, toolType, status, errorConfiguredTime)
  }
 
  const selectedValue = useMemo(() => {
    const storedValue = getFilterValue(name) || initialValue
    return storedValue as string
  }, [getFilterValue, name, initialValue])
 
  return (
    <FormControl fullWidth>
      <InputLabel size="normal" id={`${name}-select`}>
        {label}
      </InputLabel>
      <Select
        size="medium"
        placeholder={label}
        labelId={`${name}-select`}
        id={`${name}-select`}
        name={name}
        label={label}
        value={selectedValue}
        onChange={handleChange}
      >
        {options.map((option, index) => {
          return (
            <MenuItem key={`option-${index}`} value={`${option.id}`} sx={{ display: 'block' }}>
              {option?.toolType && (
                <img
                  src={option?.toolType === JENKINS ? Jenkins : CJOC}
                  alt="icon"
                  style={{ position: 'relative', top: '5px', paddingRight: '5px' }}
                />
              )}
              <Typography component="span" color={option?.status === FAILED ? 'error.main' : ''}>
                {option.name}
              </Typography>
              {option?.subName && (
                <InputLabel sx={{ fontSize: '0.8rem', cursor: 'pointer' }}>
                  {option?.subName}
                </InputLabel>
              )}
            </MenuItem>
          )
        })}
      </Select>
    </FormControl>
  )
}
 
const DashboardFilterDurationSelect = ({
  label,
  name,
  type,
  options,
  initialValue,
}: ICIInsightsFilter) => {
  const { updateFilterValue, getOptionText, getFilterValue } = useCIInsightsContext()
  const showDynamicDuration = useFeatureFlagEnabled('platform', 'showDynamicDuration')
  return (
    <DurationSelect
      label={label}
      name={name}
      type={type}
      options={options}
      initialValue={initialValue}
      showDynamicDuration={showDynamicDuration}
      updateFilterValue={updateFilterValue}
      getOptionText={getOptionText}
      getFilterValue={getFilterValue}
    />
  )
}
 
export const CIInsightsTopBar = () => {
  const [anchorEl, setAnchorEl] = useState<null | HTMLElement>(null)
  const {
    filters,
    applyFilters,
    resetFilters,
    handleFilterClose,
    appliedFilters,
    time_zone,
    displayTransition,
    layoutLoading,
  } = useCIInsightsContext()
  const { mode } = useColorMode()
  const selectedStatus = appliedFilters[CI_TOOL_ID]?.status || ''
  const selectedToolType = appliedFilters[CI_TOOL_ID]?.toolType || ''
  const selectedErrorConfiguredTime = appliedFilters[CI_TOOL_ID]?.errorConfiguredTime || ''
  const configuredTime = formatDateBasedOnTimeZone(
    selectedErrorConfiguredTime as string,
    time_zone ?? '',
  )?.toLocaleString()
 
  const handleClick = (event: React.MouseEvent<HTMLElement>) => {
    setAnchorEl(event.currentTarget)
  }
 
  const handleApplyClick = () => {
    setAnchorEl(null)
    applyFilters()
  }
 
  const handleClose = () => {
    setAnchorEl(null)
    handleFilterClose()
  }
 
  if (layoutLoading || displayTransition) return null
 
  return (
    <Stack direction="row" justifyContent="space-between" alignItems="center" padding={2}>
      <Box sx={{ display: 'flex' }}>
        <Button size="medium" variant="contained" startIcon={<FilterIcon />} onClick={handleClick}>
          {t('title')}
        </Button>
        {AppliedFilters()}
      </Box>
      {selectedStatus === FAILED && (
        <ErrorConfigurationAlert
          toolName={getToolName(selectedToolType)}
          timeStamp={configuredTime}
        />
      )}
      <Popover
        keepMounted
        open={Boolean(anchorEl)}
        anchorEl={anchorEl}
        onClose={handleClose}
        anchorOrigin={{
          vertical: 50,
          horizontal: 'left',
        }}
      >
        <Box
          width={350}
          padding={3}
          sx={{ backgroundColor: mode === 'dark' ? '#191919' : '#fafafa' }}
        >
          <FilterContent filters={filters} />
          <Box sx={{ display: 'flex', justifyContent: 'space-between' }}>
            <Button variant="text" onClick={resetFilters}>
              {t('reset')}
            </Button>
            <Button variant="contained" onClick={handleApplyClick}>
              {t('apply')}
            </Button>
          </Box>
        </Box>
      </Popover>
    </Stack>
  )
}

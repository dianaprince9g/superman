export const getTooltype = (contributionId: string) => {
  switch (true) {
    case contributionId.startsWith('cb.cjoc'):
      return CJOC
    case contributionId.startsWith('cb.jenkins'):
      return JENKINS
    case contributionId.startsWith('cb.cbci.cbci-jaas'):
      return JAAS
    case contributionId.startsWith('cb.cbci'):
      return CBCI
    default:
      return ''
  }
}
 
export const CIInsights = () => {
  const [ciIntegrations, setciIntegrations] = useState(Array<IOptions>)
 
  const { data: ciIntegrationsResponse, isLoading, isFetching } = useCIIntegrationQuery()
 
  const { timezone: time_zone, timeFormat: time_format } = usePreferences()
 
  const { canCreate } = useOrganizationPermission(RBAC_TYPES.ENDPOINT)
 
  useEffect(() => {
    if (ciIntegrationsResponse && ciIntegrationsResponse?.endpoints) {
      setciIntegrations(
        ciIntegrationsResponse?.endpoints?.map(
          (x: { id: string; name: string; contributionId: string; properties: IProperty[] }) => ({
            id: x?.id,
            name: x?.name,
            toolType: getTooltype(x?.contributionId),
            status:
              x?.properties?.find((property: { name: string }) => property?.name === STATUS)
                ?.string || NOT_INSTALLED,
            errorConfiguredTime: x?.properties?.find(
              (property: { name: string }) => property?.name === STATUS,
            )?.audit?.when,
          }),
        ),
      )
    }
  }, [ciIntegrationsResponse?.endpoints, isLoading, isFetching])
 
  const filters = useMemo(
    () => [
      {
        title: t('filters.ciIntegration.title'),
        label: t('filters.ciIntegration.label'),
        name: t('filters.ciIntegration.name'),
        type: SELECT,
        options: ciIntegrations,
        initialValue: ciIntegrations?.find(x => x)?.id,
      },
      {
        title: t('filters.duration.title'),
        label: t('filters.duration.title'),
        name: t('filters.duration.name'),
        type: DURATION_SELECT,
        options: DURATION_DROPDOWN_VALUES(),
        initialValue: DURATION_DROPDOWN_DEFAULT_VALUE,
      },
    ],
    [ciIntegrations, t],
  )
 
  if (isLoading) return <CircularProgressCenterLoader />
 
  return canCreate ? (
    <CIInsightsContextContainer
      queries={useCIInsightsWidgetsQueries}
      filters={filters as ICIInsightsFilter[]}
      widgets={ciIntegrations?.find(x => x)?.toolType === CJOC ? CJOC_WIDGETS : WIDGETS}
      widgetDrillDownQuery={useCIInsightsWidgetDrillDownQuery}
      widgetCJOCQuery={useCIInsightsCJOCWidgetQuery}
      time_zone={time_zone}
      time_format={time_format || (FORMAT_12_HR as TimeFormat)}
      ciIntegration={ciIntegrationsResponse?.endpoints}
      transition={(props: ITransition) => <Transition {...props} />}
    />
  ) : (
    <NoAccess />
  )
}

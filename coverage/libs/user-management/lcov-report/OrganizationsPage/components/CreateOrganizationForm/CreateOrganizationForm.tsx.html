import { useTranslation } from 'react-i18next'
import { boolean, object, string } from 'yup'
 
import { useGetOrganizationWithSubOrgs } from '@cloudbees/api-platform/queries/useOrganizationQueries'
import { Form } from '@cloudbees/shared-patterns/Form'
import { type IProperty } from '@cloudbees/shared-types'
 
export interface ICreateOrganizationFormValues {
  name: string
  description: string
  isOrgNameValid?: boolean
  orgDomain: string
  majorVersion: number
  minorVersion: number
}
 
interface ICreateOrganizationFormProps {
  parentOrg?: string
  children: React.ReactNode
  onSubmit?: (values: ICreateOrganizationFormValues) => void
  endpointProperties?: IProperty[]
}
 
export const CreateOrganizationForm = ({
  parentOrg,
  onSubmit,
  children,
  endpointProperties,
}: ICreateOrganizationFormProps) => {
  const { t } = useTranslation('user-management', { keyPrefix: 'user-organizations.form' })
  const { data: orgData } = useGetOrganizationWithSubOrgs(parentOrg ?? '', undefined, !!parentOrg)
 
  const nameValidation = parentOrg
    ? string()
        .required(t('name-required'))
        .test('name', t('name-duplicate'), value => {
          const isNameInvalid =
            orgData?.organization?.childOrganizations?.some(org => org.displayName === value) ??
            true
 
          return !isNameInvalid
        })
    : string().required(t('name-required'))
 
  const validationFields = {
    name: nameValidation,
    description: string(),
    majorVersion: string()
      .required(t('major-version-required'))
      .test('empty-or-num', t('version-number-required'), value => !isNaN(+value)),
    minorVersion: string()
      .required(t('minor-version-required'))
      .test('empty-or-num', t('version-number-required'), value => !isNaN(+value)),
  }
 
  const validationSchema = object(
    !parentOrg
      ? { ...validationFields, isOrgNameValid: boolean().oneOf([true], t('name-unavailable')) }
      : validationFields,
  )
 
  const major = endpointProperties?.find(property => property.name === 'major')?.int32 || 0
  const minor = endpointProperties?.find(property => property.name === 'minor')?.int32 || 0
 
  return (
    <Form.Context<ICreateOrganizationFormValues>
      initialValues={{
        name: '',
        description: '',
        isOrgNameValid: !parentOrg && false,
        orgDomain: '',
        majorVersion: major,
        minorVersion: minor,
      }}
      onSubmit={(values, { resetForm }) => {
        onSubmit && onSubmit(values)
        setTimeout(() => {
          resetForm()
        }, 5000)
      }}
      validationSchema={validationSchema}
      validateOnBlur={false}
    >
      <Form.Form>{children}</Form.Form>
    </Form.Context>
  )
}

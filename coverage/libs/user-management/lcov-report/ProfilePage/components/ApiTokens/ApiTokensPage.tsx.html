import { useState } from 'react'
import { useTranslation } from 'react-i18next'
import Box from '@mui/material/Box'
import Button from '@mui/material/Button'
import Stack from '@mui/material/Stack'
import { useQueryClient } from '@tanstack/react-query'
import addDays from 'date-fns/addDays'
import { object, string } from 'yup'
 
import { useApiMutation } from '@cloudbees/api-platform/queries/guiderails'
import {
  type IAPITokenResponse,
  type IAPITokensResponse,
  useApiTokens,
} from '@cloudbees/api-platform/queries/useApiTokensQuery'
import { useInstrumentation } from '@cloudbees/api-platform/queries/useInstrumentation'
import { useNotification } from '@cloudbees/shared-hooks/useNotification'
import { Filter } from '@cloudbees/shared-patterns/Filter'
import { Form } from '@cloudbees/shared-patterns/Form'
 
import { usePreferences } from '../../../Preferences/context/PreferencesContext'
 
import { type IAPITokenFormValues, APITokenDrawer } from './APITokenDrawer'
import { ApiTokenModal } from './ApiTokenModal'
import { ApiTokensFilters } from './ApiTokensFilters'
import { ApiTokensTable } from './ApiTokensTable'
import { DeleteApiTokenModal } from './DeleteApiTokenModal'
 
export const ApiTokensPage = () => {
  const { t } = useTranslation('user-management', {
    keyPrefix: 'user-profile.api-tokens',
  })
  const { notify } = useNotification()
  const queryClient = useQueryClient()
  const { formatDate } = usePreferences()
 
  const durationDays = (createdAt: string | undefined, expiresAt: string | undefined) => {
    const createdAtDate = createdAt ? formatDate(createdAt, 'date') : ''
    const expiresAtDate = expiresAt ? formatDate(expiresAt, 'date') : ''
    const days =
      createdAtDate && expiresAtDate
        ? Math.floor(
            (new Date(expiresAtDate).getTime() - new Date(createdAtDate).getTime()) /
              (1000 * 60 * 60 * 24),
          )
        : ''
    return days
      ? days > 1
        ? t('duration-days', {
            days,
          })
        : t('one-day')
      : ''
  }
 
  const { data } = useApiTokens()
  const [APITokenDrawerState, openAPITokenDrawer] = useState(false)
  const [ApiTokenModalState, openApiTokenModal] = useState(false)
  const [token, setToken] = useState('')
  const [tokenToDelete, setTokenToDelete] = useState<string | undefined>()
 
  const initialValues = {
    name: '',
    description: '',
    duration: '7',
    expiresAt: '',
    customDuration: '',
  }
 
  const validationSchema = object({
    name: string()
      .required(t('drawer-to-generate.name_required'))
      .matches(/^[a-z0-9][a-z0-9 ]*$/i, t('drawer-to-generate.name_invalid')),
    description: string(),
    duration: string().required(t('drawer-to-generate.duration_required')),
    customDuration: string().when(['duration', 'expiresAt'], ([duration, expiresAt]) => {
      return duration === 'custom' && !expiresAt
        ? string().required(t('drawer-to-generate.duration_required'))
        : string().notRequired()
    }),
  })
 
  const { DELETE } = useApiMutation()
  const { track } = useInstrumentation()
  const { POST } = useApiMutation<IAPITokenResponse>()
 
  const onGenerate = async (values: IAPITokenFormValues, resetForm: () => void) => {
    try {
      const { duration, ...valuesRest } = values
 
      const apiTokenValues = {
        ...valuesRest,
        expiresAt:
          duration === 'custom'
            ? valuesRest.expiresAt
            : addDays(new Date(), parseInt(duration)).toISOString(),
      }
 
      const response = await POST({ path: `/v1/api-tokens`, body: apiTokenValues })
      track('API Token Created', {
        properties: {
          expirationDate: apiTokenValues.expiresAt,
        },
      })
      setToken(response.token)
      queryClient.invalidateQueries({ queryKey: ['api-tokens'] })
      notify(t('success-messages.create_token'), 'success')
      openApiTokenModal(true)
    } catch (e) {
      notify(t('error_messages.create_token'), 'error')
    } finally {
      resetForm()
      openAPITokenDrawer(false)
    }
  }
 
  const handleDelete = async () => {
    try {
      await DELETE(`/v1/api-tokens/${tokenToDelete}`)
      queryClient.setQueryData(['api-tokens'], (oldData: IAPITokensResponse | undefined) => ({
        apiTokens: oldData?.apiTokens.filter(apiToken => apiToken.id !== tokenToDelete) || [],
      }))
      notify(t('success-messages.delete_token'), 'success')
    } catch (e) {
      notify(t('error_messages.delete_token'), 'error')
    }
  }
 
  const apiTokens =
    data?.apiTokens.map(token => {
      const createdAt = token.createdAt ?? ''
      const expiresAt = token.expiresAt ?? ''
      const duration = durationDays(createdAt, expiresAt)
 
      return {
        ...token,
        duration,
        createdAt,
        expiresAt,
      }
    }) ?? []
 
  return (
    <Filter.Container data={apiTokens} density="compact">
      <Stack className="Filters">
        <Stack px={2} pt={2} direction="row" gap={12} justifyContent="space-between">
          <ApiTokensFilters />
          <Box>
            <Button
              data-testid="action-button"
              variant="contained"
              onClick={() => openAPITokenDrawer(true)}
            >
              {t('action-button')}
            </Button>
          </Box>
        </Stack>
        <Stack sx={{ px: 2 }} direction="row" gap={1}></Stack>
      </Stack>
 
      <ApiTokensTable
        openDeleteModal={(id: string) => {
          setTokenToDelete(id)
        }}
      />
      <Form.Context
        initialValues={initialValues}
        validationSchema={validationSchema}
        onSubmit={(values: IAPITokenFormValues, { resetForm }) => {
          onGenerate(values, resetForm)
        }}
        enableReinitialize
      >
        <APITokenDrawer open={APITokenDrawerState} onClose={() => openAPITokenDrawer(false)} />
      </Form.Context>
 
      <ApiTokenModal
        open={ApiTokenModalState}
        token={token}
        onClose={() => {
          openApiTokenModal(false)
          setToken('')
        }}
      />
      <DeleteApiTokenModal
        open={!!tokenToDelete}
        onClose={() => setTokenToDelete(undefined)}
        onDelete={handleDelete}
      />
    </Filter.Container>
  )
}

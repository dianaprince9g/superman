import { useTranslation } from 'react-i18next'
import { useSearchParams } from 'react-router-dom'
import Divider from '@mui/material/Divider'
import Grid from '@mui/material/Grid'
import Typography from '@mui/material/Typography'
 
import { PageHeader } from '@cloudbees/shared-patterns/PageHeader'
import { SidebarLayout } from '@cloudbees/shared-patterns/SidebarLayout'
import { Tabs } from '@cloudbees/shared-patterns/Tabs'
import { useFeatureFlagEnabled } from '@cloudbees/shared-rox/useFeatureFlagEnabled'
 
import { ApiTokensPage } from './components/ApiTokens/ApiTokensPage'
import { AvatarProfile } from './components/Avatar'
import { LinkAccountModal } from './components/LinkAccountModal'
import { MyAccountOrganizationsTable } from './components/OrganizationsTable'
import { LinkedAccountsContent } from './LinkedAccountsContent'
import { ProfilePageContent } from './ProfilePageContent'
 
export const ProfilePage = () => {
  // A lot of this stuff is duplicated from the account security page
  const { t } = useTranslation('user-management')
  const [searchParams] = useSearchParams()
  const newAccount = searchParams.get('linkAccount')
  const verificationCode = searchParams.get('verificationCode')
  const showApiTokenTab = useFeatureFlagEnabled('platform', 'showApiTokens')
 
  // TODO: eventually the platform UserService and Keycloak will support some kind of account linking process.
  // When this is available, we need to switch this to use the UserService and Keycloak to get the linked accounts.
  const linkedAccounts: string[] = []
 
  return (
    <>
      <PageHeader.Wrapper showBreadcrumb={false}>
        <PageHeader.Title>{t('user-profile.header.title')}</PageHeader.Title>
        <PageHeader.Description>{t('user-profile.header.description')}</PageHeader.Description>
      </PageHeader.Wrapper>
      <SidebarLayout.Padded>
        <Grid
          data-testid="user-profile-page"
          container
          rowSpacing={1}
          columnSpacing={{ xs: 1, sm: 2, md: 3 }}
        >
          <Grid item xs={12}>
            <AvatarProfile />
          </Grid>
 
          <Grid item xs={12}>
            <Tabs.Wrapper tabPanelSx={{ p: 0 }}>
              <Tabs.Tab title={t('user-profile.tabs.profile')}>
                <ProfilePageContent />
              </Tabs.Tab>
              <Tabs.Tab title={t('user-profile.tabs.organizations')}>
                <Grid item xs={12}>
                  <Typography variant="h5" sx={{ mt: 2 }}>
                    {t('user-organizations.header.my-orgs')}
                  </Typography>
                  <Divider sx={{ my: 1 }} />
                  <MyAccountOrganizationsTable hideSuborgs />
                </Grid>
              </Tabs.Tab>
              {linkedAccounts.length && (
                <Tabs.Tab title={t('user-profile.tabs.linked_accounts')}>
                  <LinkedAccountsContent linkedAccounts={linkedAccounts} />
                </Tabs.Tab>
              )}
              {showApiTokenTab && (
                <Tabs.Tab title={t('user-profile.tabs.api_tokens')}>
                  <ApiTokensPage />
                </Tabs.Tab>
              )}
            </Tabs.Wrapper>
          </Grid>
 
          {newAccount ? (
            <LinkAccountModal
              email={newAccount || ''}
              linkedAccounts={linkedAccounts}
              isLoading={false}
              verificationCode={verificationCode || ''}
            />
          ) : null}
        </Grid>
      </SidebarLayout.Padded>
    </>
  )
}
